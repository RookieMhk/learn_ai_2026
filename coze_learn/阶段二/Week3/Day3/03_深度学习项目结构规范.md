**好处：**
- 🔍 **可定位**：每个功能在固定位置
- 🔄 **可复用**：模块可被多个脚本引用
- 🤝 **可协作**：团队成员快速理解项目
- 📦 **可部署**：训练、验证、推理解耦
- 🧪 **可实验**：配置文件管理超参数

#### (2) `train.py` - 训练脚本

```python
"""
训练入口脚本
"""
import torch
import torch.nn as nn
from torch.utils.data import DataLoader
from config import Config
from models import build_model
from data import build_dataset
from optim import build_optimizer, build_scheduler
from utils import setup_logger, set_seed, save_checkpoint

def train(cfg):
    # 设置随机种子
    set_seed(cfg.seed)
    
    # 设置日志
    logger = setup_logger(cfg)
    
    # 设备
    device = torch.device(cfg.device)
    logger.info(f"Using device: {device}")
    
    # 数据
    train_dataset = build_dataset(cfg, mode='train')
    val_dataset = build_dataset(cfg, mode='val')
    train_loader = DataLoader(train_dataset, batch_size=cfg.batch_size, 
                               shuffle=True, num_workers=cfg.num_workers)
    val_loader = DataLoader(val_dataset, batch_size=cfg.batch_size,
                            shuffle=False, num_workers=cfg.num_workers)
    
    # 模型
    model = build_model(cfg).to(device)
    
    # 优化器
    optimizer = build_optimizer(model, cfg)
    scheduler = build_scheduler(optimizer, cfg)
    
    # 损失函数
    criterion = nn.CrossEntropyLoss()
    
    # 训练循环
    best_acc = 0
    for epoch in range(cfg.epochs):
        # 训练
        train_loss, train_acc = train_epoch(model, train_loader, 
                                            criterion, optimizer, device)
        
        # 验证
        val_loss, val_acc = validate(model, val_loader, 
                                      criterion, device)
        
        scheduler.step()
        
        # 保存检查点
        if val_acc > best_acc:
            best_acc = val_acc
            save_checkpoint(model, optimizer, epoch, val_acc, cfg.output_dir)
        
        logger.info(f"Epoch {epoch}: Train Acc={train_acc:.4f}, Val Acc={val_acc:.4f}")

    cfg = Config.from_yaml("config/base.yaml")
```

```python
"""
模型注册机制，方便通过名称创建模型
"""
from .resnet import ResNet18, ResNet34, ResNet50
from .transformer import VisionTransformer

__models__ = {
    'resnet18': ResNet18,
    'resnet34': ResNet34,
    'resnet50': ResNet50,
    'vit': VisionTransformer,
}

def build_model(cfg):
    model_name = cfg.model.name.lower()
    if model_name not in __models__:
        raise ValueError(f"Unknown model: {model_name}")
    return __models__[model_name](**cfg.model)

__all__ = list(__models__.keys()) + ['build_model']
```

## 三、配置文件管理

### 3.1 配置类设计

```python
from dataclasses import dataclass
from typing import Optional
import yaml

@dataclass
class ModelConfig:
    name: str = "resnet50"
    pretrained: bool = True
    num_classes: int = 1000
    dropout: float = 0.0

@dataclass
class TrainConfig:
    epochs: int = 100
    lr: float = 0.001
    batch_size: int = 32
    optimizer: str = "adam"
    scheduler: str = "cosine"

class Config:
    def __init__(self, yaml_path: Optional[str] = None):
        self.model = ModelConfig()
        self.train = TrainConfig()
        
        if yaml_path:
            self.load(yaml_path)
    
    def load(self, yaml_path):
        with open(yaml_path, 'r') as f:
            config_dict = yaml.safe_load(f)
        
        # 更新配置
        if 'model' in config_dict:
            for k, v in config_dict['model'].items():
                setattr(self.model, k, v)
        if 'train' in config_dict:
            for k, v in config_dict['train'].items():
                setattr(self.train, k, v)
    
    @classmethod
    def from_yaml(cls, yaml_path):
        return cls(yaml_path)
```

### 4.1 日志配置

```python
import logging
import os
from datetime import datetime

def setup_logger(cfg):
    # 创建输出目录
    log_dir = os.path.join(cfg.output_dir, "logs")
    os.makedirs(log_dir, exist_ok=True)
    
    # 日志文件名
    timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
    log_file = os.path.join(log_dir, f"train_{timestamp}.log")
    
    # 配置logger
    logger = logging.getLogger("train")
    logger.setLevel(logging.INFO)
    
    # 文件handler
    fh = logging.FileHandler(log_file)
    fh.setLevel(logging.INFO)
    
    # 控制台handler
    ch = logging.StreamHandler()
    ch.setLevel(logging.INFO)
    
    # 格式化
    formatter = logging.Formatter(
        '%(asctime)s - %(name)s - %(levelname)s - %(message)s'
    )
    fh.setFormatter(formatter)
    ch.setFormatter(formatter)
    
    logger.addHandler(fh)
    logger.addHandler(ch)
    
    return logger
```

## 五、检查点管理

### 5.1 保存与加载

```python
import torch
import os

def save_checkpoint(model, optimizer, epoch, metrics, save_dir, filename="checkpoint.pth"):
    os.makedirs(save_dir, exist_ok=True)
    
    checkpoint = {
        'epoch': epoch,
        'model_state_dict': model.state_dict(),
        'optimizer_state_dict': optimizer.state_dict(),
        'metrics': metrics,
    }
    
    filepath = os.path.join(save_dir, filename)
    torch.save(checkpoint, filepath)
    print(f"Checkpoint saved to {filepath}")

def load_checkpoint(model, optimizer, checkpoint_path):
    checkpoint = torch.load(checkpoint_path)
    
    model.load_state_dict(checkpoint['model_state_dict'])
    if optimizer is not None:
        optimizer.load_state_dict(checkpoint['optimizer_state_dict'])
    
    epoch = checkpoint['epoch']
    metrics = checkpoint['metrics']
    
    print(f"Checkpoint loaded from {checkpoint_path}")
    return epoch, metrics
```

### 5.2 自动保存最佳模型

```python
class CheckpointManager:
    def __init__(self, save_dir, metric_name='accuracy', mode='max'):
        self.save_dir = save_dir
        self.metric_name = metric_name
        self.mode = mode
        self.best_value = float('-inf') if mode == 'max' else float('inf')
        
    def should_save(self, metrics):
        current_value = metrics.get(self.metric_name)
        if current_value is None:
            return False
        
        if self.mode == 'max':
            return current_value > self.best_value
        else:
            return current_value < self.best_value
    
    def save_if_best(self, model, optimizer, epoch, metrics):
        if self.should_save(metrics):
            self.best_value = metrics[self.metric_name]
            save_checkpoint(model, optimizer, epoch, metrics, 
                          self.save_dir, "best_model.pth")
            print(f"Best {self.metric_name}: {self.best_value:.4f}")
```

## 六、数据管理

### 6.1 数据集模板

```python
from torch.utils.data import Dataset
import os
import torch
from PIL import Image

class BaseDataset(Dataset):
    """数据集基类"""
    
    def __init__(self, data_dir, split='train', transform=None):
        self.data_dir = data_dir
        self.split = split
        self.transform = transform
        
        self.samples = self._load_samples()
    
    def _load_samples(self):
        """加载样本列表，子类实现"""
        raise NotImplementedError
    
    def __len__(self):
        return len(self.samples)
    
    def __getitem__(self, idx):
        img_path, label = self.samples[idx]
        
        # 加载图像
        image = Image.open(img_path).convert('RGB')
        
        # 数据增强
        if self.transform:
            image = self.transform(image)
        
        return image, label

class MyDataset(BaseDataset):
    def _load_samples(self):
        samples = []
        split_file = os.path.join(self.data_dir, f"{self.split}.txt")
        
        with open(split_file, 'r') as f:
            for line in f:
                img_name, label = line.strip().split()
                img_path = os.path.join(self.data_dir, 'images', img_name)
                samples.append((img_path, int(label)))
        
        return samples
```
