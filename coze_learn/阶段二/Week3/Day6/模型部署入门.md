## 2. ONNX 简介与导出 <a name="2-onnx-简介与导出"></a>

### 2.1 什么是 ONNX？

**ONNX (Open Neural Network Exchange)** 是一种开放的模型表示标准：

```
┌─────────────┐         ┌─────────────┐         ┌─────────────┐
│   PyTorch   │  ──>    │    ONNX     │  ──>    │  ONNX       │
│   模型      │  导出    │    模型     │  加载    │  Runtime    │
└─────────────┘         └─────────────┘         └─────────────┘
      训练                   中间格式              推理
```

### 2.2 ONNX 的优势

| 优势 | 说明 |
|------|------|
| **框架无关** | 任何框架训练的模型都可以导出 |
| **跨平台** | Windows, Linux, macOS, Mobile |
| **硬件优化** | 支持 GPU、CPU、专用加速器 |
| **生态丰富** | 众多部署工具支持 |

## 3. PyTorch 模型转 ONNX <a name="3-pytorch-模型转-onnx"></a>

### 3.1 基本导出方法

```python
import torch
import torchvision.models as models

# 1. 创建模型
model = models.resnet18(pretrained=True)
model.eval()  # 切换到推理模式

# 2. 准备示例输入
dummy_input = torch.randn(1, 3, 224, 224)

# 3. 导出为 ONNX
torch.onnx.export(
    model,                           # 模型
    dummy_input,                     # 示例输入
    "resnet18.onnx",                 # 输出文件
    export_params=True,             # 导出参数
    opset_version=11,               # ONNX 算子集版本
    do_constant_folding=True,       # 常量折叠优化
    input_names=['input'],          # 输入名称
    output_names=['output'],        # 输出名称
    dynamic_axes={                  # 动态维度
        'input': {0: 'batch_size'},
        'output': {0: 'batch_size'}
    }
)

print("✅ 模型已导出至 resnet18.onnx")
```

### 3.2 验证导出的 ONNX 模型

```python
import onnx
import onnxruntime as ort

# 1. 加载模型
onnx_model = onnx.load("resnet18.onnx")

# 2. 检查模型完整性
onnx.checker.check_model(onnx_model)
print("✅ ONNX 模型格式检查通过")

# 3. 打印模型信息
print(f"\n📊 模型信息:")
print(f"   输入: {[inp.name for inp in onnx_model.graph.input]}")
print(f"   输出: {[out.name for out in onnx_model.graph.output]}")

# 4. 使用 ONNX Runtime 推理验证
session = ort.InferenceSession("resnet18.onnx")
input_name = session.get_inputs()[0].name
output_name = session.get_outputs()[0].name

# 准备输入数据
test_input = torch.randn(1, 3, 224, 224).numpy()

# ONNX Runtime 推理
onnx_output = session.run([output_name], {input_name: test_input})[0]

# PyTorch 推理
with torch.no_grad():
    pytorch_input = torch.from_numpy(test_input)
    pytorch_output = model(pytorch_input).numpy()

# 对比结果
diff = np.abs(onnx_output - pytorch_output).max()
print(f"\n📈 推理结果最大差异: {diff:.6f}")
if diff < 1e-5:
    print("✅ PyTorch 和 ONNX 推理结果一致!")
```

### 4.2 基本推理代码

```python
import numpy as np
import onnxruntime as ort

class ONNXInference:
    """ONNX 模型推理封装"""
    
    def __init__(self, model_path, providers=None):
        """
        初始化推理会话
        
        Args:
            model_path: ONNX 模型文件路径
            providers: 推理提供者 ['CPUExecutionProvider', 'CUDAExecutionProvider']
        """
        if providers is None:
            providers = ['CPUExecutionProvider']
            
        self.session = ort.InferenceSession(model_path, providers=providers)
        self.input_name = self.session.get_inputs()[0].name
        self.output_name = self.session.get_outputs()[0].name
        
        # 获取模型信息
        self.input_shape = self.session.get_inputs()[0].shape
        self.input_type = self.session.get_inputs()[0].type
        
        print(f"✅ 模型加载成功")
        print(f"   输入: {self.input_name}, shape={self.input_shape}, type={self.input_type}")
        print(f"   输出: {self.output_name}")
        print(f"   Provider: {self.session.get_providers()}")
        
    def predict(self, input_data):
        """
        执行推理
        
        Args:
            input_data: numpy array, shape = (batch, channels, height, width)
            
        Returns:
            numpy array: 模型输出
        """
        return self.session.run(
            [self.output_name], 
            {self.input_name: input_data.astype(np.float32)}
        )[0]
    
    def predict_batch(self, data_list, batch_size=32):
        """
        批量推理
        
        Args:
            data_list: 数据列表
            batch_size: 批次大小
            
        Returns:
            list: 所有预测结果
        """
        results = []
        for i in range(0, len(data_list), batch_size):
            batch = np.stack(data_list[i:i+batch_size])
            batch_results = self.predict(batch)
            results.extend(batch_results)
        return results


# 使用示例
if __name__ == "__main__":
    # 加载模型
    inferencer = ONNXInference(
        "resnet18.onnx",
        providers=['CPUExecutionProvider']
    )
    
    # 准备输入 (示例)
    dummy_input = np.random.randn(1, 3, 224, 224).astype(np.float32)
    
    # 推理
    output = inferencer.predict(dummy_input)
    print(f"输出 shape: {output.shape}")
    print(f"预测类别: {np.argmax(output)}")
```

### 4.3 性能基准测试

```python
import time

def benchmark_onnx(model_path, input_shape=(1, 3, 224, 224), 
                   n_warmup=10, n_runs=100):
    """
    ONNX 模型性能基准测试
    """
    # 创建推理器
    session = ort.InferenceSession(model_path)
    input_name = session.get_inputs()[0].name
    
    # 准备输入
    dummy_input = np.random.randn(*input_shape).astype(np.float32)
    
    # 预热
    print("预热中...")
    for _ in range(n_warmup):
        session.run([session.get_outputs()[0].name], 
                    {input_name: dummy_input})
    
    # 基准测试
    print("运行基准测试...")
    times = []
    
    for _ in range(n_runs):
        start = time.perf_counter()
        session.run([session.get_outputs()[0].name], 
                   {input_name: dummy_input})
        end = time.perf_counter()
        times.append((end - start) * 1000)  # 毫秒
    
    times = np.array(times)
    
    print(f"\n📊 性能基准测试结果 ({n_runs} 次运行):")
    print(f"   平均延迟: {times.mean():.2f} ms")
    print(f"   中位数延迟: {np.median(times):.2f} ms")
    print(f"   最小延迟: {times.min():.2f} ms")
    print(f"   最大延迟: {times.max():.2f} ms")
    print(f"   P95 延迟: {np.percentile(times, 95):.2f} ms")
    print(f"   P99 延迟: {np.percentile(times, 99):.2f} ms")
    print(f"   吞吐量: {1000/times.mean():.2f} FPS")
    
    return times

# 运行基准测试
benchmark_onnx("resnet18.onnx")
```

## 5. 模型优化技术 <a name="5-模型优化技术"></a>

### 5.1 模型量化 (Quantization)

将 FP32 转换为 INT8，减少模型大小和加速推理：

```python
# 方法1: 动态量化 (简单，推荐入门)
import torch.quantization

# 动态量化 (权重int8， activations fp32)
quantized_model = torch.quantization.quantize_dynamic(
    model,                        # 原始模型
    {torch.nn.Linear},            # 要量化的层类型
    dtype=torch.qint8            # 量化精度
)

# 保存量化模型
torch.save(quantized_model.state_dict(), 'model_quantized.pth')

# 导出为 ONNX
torch.onnx.export(
    quantized_model,
    dummy_input,
    "model_quantized.onnx",
    input_names=['input'],
    output_names=['output']
)
```

```python
# 方法2: 静态量化 (更高压缩率，需要校准数据)
```

### 5.2 模型剪枝 (Pruning)

移除不重要的连接：

```python
import torch.nn.utils.prune as prune

def prune_model(model, amount=0.3):
    """
    剪枝模型
    
    Args:
        model: 要剪枝的模型
        amount: 剪枝比例 (0.3 = 移除30%的连接)
    """
    for name, module in model.named_modules():
        if isinstance(module, torch.nn.Conv2d):
            # L1 非结构化剪枝
            prune.l1_unstructured(module, name='weight', amount=amount)
            prune.remove(module, 'weight')  # 永久化剪枝
            
        elif isinstance(module, torch.nn.Linear):
            prune.l1_unstructured(module, name='weight', amount=amount)
            prune.remove(module, 'weight')
            
    return model

# 使用
pruned_model = prune_model(model, amount=0.5)
```

### 5.3 知识蒸馏 (Knowledge Distillation)

大模型指导小模型学习：

```python
def knowledge_distillation(student_model, teacher_model, 
                          train_loader, temperature=4.0, alpha=0.7):
    """
    知识蒸馏训练
    
    Args:
        student_model: 学生模型 (小)
        teacher_model: 教师模型 (大)
        train_loader: 训练数据
        temperature: 温度参数，越大越软化
        alpha: 损失权重 (alpha * soft_loss + (1-alpha) * hard_loss)
    """
    optimizer = torch.optim.Adam(student_model.parameters(), lr=0.001)
    criterion = nn.KLDivLoss(reduction='batchmean')
    
    student_model.train()
    teacher_model.eval()  # 教师模型不更新
    
    for inputs, targets in train_loader:
        optimizer.zero_grad()
        
        # 教师输出 (软标签)
        with torch.no_grad():
            teacher_logits = teacher_model(inputs)
            teacher_probs = F.softmax(teacher_logits / temperature, dim=1)
            
        # 学生输出
        student_logits = student_model(inputs)
        student_log_probs = F.log_softmax(student_logits / temperature, dim=1)
        
        # 软损失 (知识蒸馏损失)
        soft_loss = criterion(student_log_probs, teacher_probs) * (temperature ** 2)
        
        # 硬损失 (普通交叉熵)
        hard_loss = F.cross_entropy(student_logits, targets)
        
        # 总损失
        loss = alpha * soft_loss + (1 - alpha) * hard_loss
        
        loss.backward()
        optimizer.step()
```

## 6. 部署实战案例 <a name="6-部署实战案例"></a>

### 6.1 完整的导出与推理流程

```python
"""
完整的模型导出与推理流程
"""

import torch
import torch.nn as nn
import torch.nn.functional as F
import numpy as np
import onnxruntime as ort
from PIL import Image
import torchvision.transforms as transforms

# ============ 1. 定义并训练模型 ============
class SimpleCNN(nn.Module):
    def __init__(self, num_classes=10):
        super().__init__()
        self.features = nn.Sequential(
            nn.Conv2d(3, 32, 3, padding=1),
            nn.ReLU(),
            nn.MaxPool2d(2),
            nn.Conv2d(32, 64, 3, padding=1),
            nn.ReLU(),
            nn.MaxPool2d(2),
            nn.Conv2d(64, 64, 3, padding=1),
            nn.ReLU(),
            nn.AdaptiveAvgPool2d(1)
        )
        self.classifier = nn.Linear(64, num_classes)
        
    def forward(self, x):
        x = self.features(x)
        x = x.view(x.size(0), -1)
        x = self.classifier(x)
        return x

# 训练代码省略，假设已有训练好的模型
model = SimpleCNN(num_classes=10)
model.eval()

# ============ 2. 导出为 ONNX ============
dummy_input = torch.randn(1, 3, 32, 32)

torch.onnx.export(
    model,
    dummy_input,
    "simple_cnn.onnx",
    input_names=['input'],
    output_names=['output'],
    dynamic_axes={'input': {0: 'batch_size'}, 'output': {0: 'batch_size'}},
    opset_version=12
)

print("✅ 模型已导出")

# ============ 3. 使用 ONNX Runtime 推理 ============
# 加载模型
session = ort.InferenceSession("simple_cnn.onnx")

# 图像预处理
transform = transforms.Compose([
    transforms.Resize((32, 32)),
    transforms.ToTensor(),
    transforms.Normalize((0.5, 0.5, 0.5), (0.5, 0.5, 0.5))
])

def preprocess_image(image_path):
    """图像预处理"""
    img = Image.open(image_path).convert('RGB')
    img_tensor = transform(img).unsqueeze(0).numpy()
    return img_tensor

def inference(image_path):
    """端到端推理"""
    # 预处理
    input_data = preprocess_image(image_path)
    
    # 推理
    output = session.run(None, {'input': input_data})[0]
    
    # 后处理
    predicted_class = np.argmax(output)
    confidence = F.softmax(torch.from_numpy(output), dim=1).max().item()
    
    return predicted_class, confidence

@app.route('/predict', methods=['POST'])

    # 获取图像数据
    image_data = request.files['image'].read()
    
    # 预处理
    # ... (图像处理代码)
    
    # 推理
    output = session.run(None, {'input': input_data})[0]
    
    # 返回结果
    return jsonify({
        'class': int(np.argmax(output)),
        'confidence': float(output.max())
    })
```
