## 案例1：Scikit-learn Pipeline自动化特征工程流程

### 1.1 问题场景

**业务背景**：某电商平台需要构建用户购买预测模型，预测用户是否会购买某商品。数据包含用户历史行为、商品属性、时间信息等。

**数据特点**：
- 特征类型多样：数值型、类别型、时间型
- 存在缺失值和异常值
- 特征间存在交互效应
- 类别不均衡（购买:不购买 ≈ 1:10）

**传统方法痛点**：
1. 特征工程步骤繁琐，手动操作容易出错
2. 预处理和特征工程代码分散，难以复现
3. 特征选择依赖经验，缺乏系统性方法
4. 模型调优和特征工程分离，难以协同优化

### 1.2 工具选择：Scikit-learn Pipeline

**选择理由**：
1. **统一接口**：将所有处理步骤封装为统一接口
2. **可复现性**：确保数据处理流程完全一致
3. **避免数据泄露**：自动处理训练集和测试集分割
4. **便于调优**：与超参数优化工具无缝集成

**Pipeline设计原则**：
```
原始数据 → 数据预处理 → 特征构造 → 特征选择 → 模型训练 → 评估
```

```python

# 案例1：电商用户购买预测Pipeline
# ============================================================================

import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns
from sklearn.datasets import make_classification
from sklearn.model_selection import train_test_split, cross_val_score, GridSearchCV

# 设置中文字体

plt.rcParams['axes.unicode_minus'] = False

def generate_ecommerce_data(n_samples=5000):
    """
    生成模拟电商用户行为数据
    """
    # 生成基础特征
    X, y = make_classification(
        n_samples=n_samples,
        n_features=15,
        n_informative=8,
        n_redundant=3,
        n_repeated=2,
        n_clusters_per_class=2,
        flip_y=0.1,
        class_sep=0.8,
        weights=[0.9, 0.1],  # 类别不均衡
        random_state=42
    )
    
    # 创建DataFrame
    feature_names = [
        'user_age', 'user_income', 'browse_duration', 'page_views',
        'cart_items', 'wishlist_count', 'session_count', 'avg_order_value',
        'days_since_last_purchase', 'product_price', 'product_rating',
        'discount_rate', 'competitor_price', 'seasonality', 'weekend_flag'
    ]
    
    df = pd.DataFrame(X, columns=feature_names)
    df['target'] = y
    
    # 添加分类特征
    df['user_gender'] = np.random.choice(['M', 'F'], size=n_samples)
    df['user_tier'] = np.random.choice(['bronze', 'silver', 'gold', 'platinum'], size=n_samples)
    df['device_type'] = np.random.choice(['mobile', 'desktop', 'tablet'], size=n_samples)
    
    # 添加缺失值（模拟真实数据）
    for col in ['user_income', 'avg_order_value', 'product_rating']:
        mask = np.random.rand(n_samples) < 0.05
        df.loc[mask, col] = np.nan
    
    # 添加异常值
    outlier_mask = np.random.rand(n_samples) < 0.02
    df.loc[outlier_mask, 'browse_duration'] = df.loc[outlier_mask, 'browse_duration'] * 10
    
    return df

# 2. 数据准备

df = generate_ecommerce_data(5000)

print(f"类别分布:\n{df['target'].value_counts(normalize=True)}")

X = df.drop('target', axis=1)
y = df['target']

# 划分训练测试集
X_train, X_test, y_train, y_test = train_test_split(
    X, y, test_size=0.2, stratify=y, random_state=42
)

print(f"训练集: {X_train.shape}, 测试集: {X_test.shape}")

# 3. 定义特征类型
numeric_features = [
    'user_age', 'user_income', 'browse_duration', 'page_views',
    'cart_items', 'wishlist_count', 'session_count', 'avg_order_value',
    'days_since_last_purchase', 'product_price', 'product_rating',
    'discount_rate', 'competitor_price', 'seasonality', 'weekend_flag'
]

categorical_features = ['user_gender', 'user_tier', 'device_type']

# 4. 构建完整Pipeline
# 数值型特征处理管道
numeric_transformer = Pipeline(steps=[
    ('imputer', SimpleImputer(strategy='median')),  # 缺失值填充
    ('scaler', StandardScaler()),  # 标准化
    ('variance_threshold', VarianceThreshold(threshold=0.01)),  # 低方差过滤
    ('poly', PolynomialFeatures(degree=2, interaction_only=True)),  # 交互特征
])

# 类别型特征处理管道
categorical_transformer = Pipeline(steps=[
    ('imputer', SimpleImputer(strategy='most_frequent')),  # 缺失值填充
    ('onehot', OneHotEncoder(handle_unknown='ignore', drop='first')),  # 独热编码
])

# 合并预处理步骤
preprocessor = ColumnTransformer(
    transformers=[
        ('num', numeric_transformer, numeric_features),
        ('cat', categorical_transformer, categorical_features)
    ]
)

# 完整Pipeline：预处理 + 特征选择 + 模型
full_pipeline = Pipeline(steps=[
    ('preprocessor', preprocessor),
    ('feature_selection', SelectKBest(score_func=f_classif, k=30)),  # 选择Top30特征
    ('classifier', RandomForestClassifier(
        n_estimators=100,
        max_depth=10,
        class_weight='balanced',
        random_state=42
    ))
])

roc_auc = roc_auc_score(y_test, y_pred_proba)

grid_search = GridSearchCV(

# 传统方法：手动特征工程
def traditional_approach(X_train, X_test, y_train, y_test):
    """传统手动特征工程方法"""
    # 1. 缺失值处理
    from sklearn.impute import SimpleImputer
    num_imputer = SimpleImputer(strategy='median')
    cat_imputer = SimpleImputer(strategy='most_frequent')
    
    X_train_num = num_imputer.fit_transform(X_train[numeric_features])
    X_test_num = num_imputer.transform(X_test[numeric_features])
    X_train_cat = cat_imputer.fit_transform(X_train[categorical_features])
    X_test_cat = cat_imputer.transform(X_test[categorical_features])
    
    # 2. 特征缩放
    from sklearn.preprocessing import StandardScaler
    scaler = StandardScaler()
    X_train_num_scaled = scaler.fit_transform(X_train_num)
    X_test_num_scaled = scaler.transform(X_test_num)
    
    # 3. 特征编码
    from sklearn.preprocessing import OneHotEncoder
    encoder = OneHotEncoder(drop='first')
    X_train_cat_encoded = encoder.fit_transform(X_train_cat).toarray()
    X_test_cat_encoded = encoder.transform(X_test_cat).toarray()
    
    # 4. 合并特征
    X_train_processed = np.hstack([X_train_num_scaled, X_train_cat_encoded])
    X_test_processed = np.hstack([X_test_num_scaled, X_test_cat_encoded])
    
    # 5. 训练模型
    from sklearn.ensemble import RandomForestClassifier
    rf = RandomForestClassifier(n_estimators=100, random_state=42)
    rf.fit(X_train_processed, y_train)
    
    # 6. 评估
    y_pred_proba = rf.predict_proba(X_test_processed)[:, 1]
    roc_auc = roc_auc_score(y_test, y_pred_proba)
    
    return roc_auc

# 运行传统方法
```

## 案例2：FeatureTools自动化特征生成

### 2.1 问题场景

**业务背景**：银行需要构建客户信用风险评估模型。数据包含客户基本信息、历史交易记录、贷款记录等多个相关联的表。

**数据特点**：
- 多表关联：客户表、交易表、贷款表等
- 时间序列特征：交易时间、还款时间等
- 层次结构：一个客户有多条交易记录
- 特征维度高：原始特征数百个

**传统方法痛点**：
1. 手动构造关联特征效率低下
2. 时间窗口特征计算复杂
3. 难以捕捉跨表特征交互
4. 特征工程过程不可复现

### 2.2 工具选择：FeatureTools

**选择理由**：
1. **深度特征合成**：自动从多表数据生成复合特征
2. **时间序列支持**：基于时间索引的特征计算
3. **关系建模**：基于实体-关系模型的特征生成
4. **可解释性**：提供特征生成路径和语义解释

### 2.3 完整代码实现

```python
# ============================================================================
# 案例2：银行信用风险评估 - FeatureTools自动化特征生成
# ============================================================================

import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns
import featuretools as ft
from sklearn.model_selection import train_test_split
from sklearn.ensemble import RandomForestClassifier
from sklearn.metrics import roc_auc_score, classification_report
import warnings
warnings.filterwarnings('ignore')

# 设置中文字体
plt.rcParams['font.sans-serif'] = ['Noto Sans CJK JP']
plt.rcParams['axes.unicode_minus'] = False

# 1. 生成模拟银行数据
def generate_bank_data(n_customers=1000, n_transactions_per_customer=50):
    """
    生成模拟银行多表数据
    """
    print("生成银行多表数据...")
    
    # 客户表
    customers = pd.DataFrame({
        'customer_id': range(n_customers),
        'age': np.random.randint(18, 70, n_customers),
        'income': np.random.lognormal(10, 0.5, n_customers),
        'employment_years': np.random.exponential(5, n_customers),
        'education': np.random.choice(['high_school', 'bachelor', 'master', 'phd'], n_customers),
        'marital_status': np.random.choice(['single', 'married', 'divorced'], n_customers),
        'has_default_history': np.random.choice([0, 1], n_customers, p=[0.85, 0.15])
    })
    
    # 交易表（每个客户多条交易记录）
    transactions_list = []
    for cust_id in range(n_customers):
        n_trans = np.random.poisson(n_transactions_per_customer)
        for i in range(n_trans):
            transactions_list.append({
                'transaction_id': len(transactions_list),
                'customer_id': cust_id,
                'amount': np.random.exponential(100),
                'transaction_type': np.random.choice(['deposit', 'withdrawal', 'transfer'], p=[0.4, 0.4, 0.2]),
                'timestamp': pd.Timestamp('2025-01-01') + pd.Timedelta(days=np.random.randint(0, 365)),
                'location': np.random.choice(['branch', 'atm', 'online'])
            })
    
    transactions = pd.DataFrame(transactions_list)
    
    # 贷款表（部分客户有贷款记录）
    loans_list = []
    for cust_id in range(n_customers):
        if np.random.rand() < 0.3:  # 30%客户有贷款
            loans_list.append({
                'loan_id': len(loans_list),
                'customer_id': cust_id,
                'loan_amount': np.random.lognormal(10, 0.3),
                'interest_rate': np.random.uniform(0.03, 0.15),
                'term_months': np.random.choice([12, 24, 36, 60]),
                'start_date': pd.Timestamp('2024-01-01') + pd.Timedelta(days=np.random.randint(0, 730)),
                'status': np.random.choice(['active', 'closed', 'defaulted'], p=[0.6, 0.35, 0.05])
            })
    
    loans = pd.DataFrame(loans_list)
    
    # 目标变量：信用风险（1=高风险，0=低风险）
    # 基于多个因素综合决定
    customers['credit_risk'] = 0
    
    # 高风险客户规则
    risk_factors = (
        (customers['age'] < 25) |
        (customers['income'] < 30000) |
        (customers['has_default_history'] == 1) |
        (customers['employment_years'] < 1)
    )
    
    customers.loc[risk_factors, 'credit_risk'] = 1
    # 添加一些噪声
    noise = np.random.rand(n_customers) < 0.05
    customers.loc[noise, 'credit_risk'] = 1 - customers.loc[noise, 'credit_risk']
    
    print(f"客户表: {customers.shape}")
    print(f"交易表: {transactions.shape}")
    print(f"贷款表: {loans.shape}")
    print(f"信用风险分布: {customers['credit_risk'].value_counts(normalize=True).to_dict()}")
    
    return customers, transactions, loans

# 2. 数据准备
print("\n" + "=" * 60)
print("案例2：银行信用风险评估 - FeatureTools自动化特征生成")

es = ft.EntitySet(id="bank_data")

# 添加客户实体
es = es.add_dataframe(
    dataframe_name="customers",
    dataframe=customers,
    index="customer_id",
    time_index=None
)

# 添加交易实体
es = es.add_dataframe(
    dataframe_name="transactions",
    dataframe=transactions,
    index="transaction_id",
    time_index="timestamp"
)

# 添加贷款实体
es = es.add_dataframe(
    dataframe_name="loans",
    dataframe=loans,
    index="loan_id",
    time_index="start_date"
)

# 4. 定义实体关系
print("\n定义实体关系...")

# 客户-交易关系（一个客户有多条交易记录）
es = es.add_relationship(
    ft.Relationship(
        es["customers"]["customer_id"],
        es["transactions"]["customer_id"]
    )
)

# 客户-贷款关系（一个客户可能有多个贷款）
es = es.add_relationship(
    ft.Relationship(
        es["customers"]["customer_id"],
        es["loans"]["customer_id"]
    )
)

print("实体集结构:")
print(es)

# 5. 深度特征合成
print("\n" + "=" * 60)
print("深度特征合成（Deep Feature Synthesis）")
print("=" * 60)

# 定义特征合成参数
feature_matrix, feature_defs = ft.dfs(
    entityset=es,
    target_dataframe="customers",
    max_depth=2,  # 特征合成深度
    agg_primitives=["mean", "std", "max", "min", "count", "sum", "trend"],
    trans_primitives=["month", "weekday", "cum_sum", "time_since_last"],
    ignore_variables={
        "customers": ["credit_risk", "has_default_history"]  # 避免目标泄露
    },
    verbose=True,
    n_jobs=-1,
    features_only=False
)

print(f"\n特征矩阵形状: {feature_matrix.shape}")
print(f"生成特征数量: {len(feature_defs)}")

# 6. 特征解释
print("\n" + "=" * 60)
print("特征生成解释")
print("=" * 60)

# 查看前10个特征的生成路径
print("\n前10个特征的生成路径:")
for i, feat in enumerate(feature_defs[:10]):
```
