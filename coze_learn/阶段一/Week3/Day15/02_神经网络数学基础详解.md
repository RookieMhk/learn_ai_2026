# 神经网络数学基础详解

## 1. 前向传播的矩阵运算

### 1.1 单层神经网络的前向传播

对于一个具有 $n$ 个输入特征、$m$ 个神经元的单层神经网络，前向传播的数学表达式为：

$$z = W x + b$$
$$a = σ(z)$$

其中：
- $x ∈ ℝⁿ$：输入向量
- $W ∈ ℝᵐ ᵗⁱᵐᵉˢ ⁿ$：权重矩阵
- $b ∈ ℝᵐ$：偏置向量
- $σ(·)$：激活函数
- $z ∈ ℝᵐ$：线性变换结果
- $a ∈ ℝᵐ$：激活输出

### 1.2 多层神经网络的前向传播

对于 $L$ 层神经网络，第 $l$ 层的前向传播公式为：

$$z[ˡ] = W[ˡ] a[ˡ⁻¹] + b[ˡ]$$
$$a[ˡ] = σ[ˡ](z[ˡ])$$

其中 $a[⁰] = x$ 为输入。

### 1.3 Python实现：前向传播

```python
import numpy as np

def initialize_parameters(layer_dims):
    """
    初始化神经网络参数
    
    参数：
    layer_dims -- 包含各层神经元数量的列表，如 [input_size, hidden1_size, ..., output_size]
    
    返回：
    parameters -- 包含权重和偏置的字典
    """
    np.random.seed(42)
    parameters = {}
    L = len(layer_dims)
    
    for l in range(1, L):
        # He初始化，适用于ReLU激活函数
        parameters[f'W{l}'] = np.random.randn(layer_dims[l], layer_dims[l-1]) * np.sqrt(2.0 / layer_dims[l-1])
        parameters[f'b{l}'] = np.zeros((layer_dims[l], 1))
    
    return parameters

def linear_forward(A_prev, W, b):
    """
    单层线性前向传播
    
    参数：
    A_prev -- 上一层激活输出，形状为 (n_prev, m)
    W -- 权重矩阵，形状为 (n_current, n_prev)
    b -- 偏置向量，形状为 (n_current, 1)
    
    返回：
    Z -- 线性变换结果
    cache -- 包含A_prev, W, b的元组，用于反向传播
    """
    Z = np.dot(W, A_prev) + b
    cache = (A_prev, W, b)
    return Z, cache

def sigmoid(Z):
    """Sigmoid激活函数"""
    return 1 / (1 + np.exp(-Z)), Z

def relu(Z):
    """ReLU激活函数"""
    return np.maximum(0, Z), Z

def linear_activation_forward(A_prev, W, b, activation):
    """
    单层线性+激活前向传播
    
    参数：
    A_prev -- 上一层激活输出
    W, b -- 当前层参数
    activation -- 激活函数名称，'sigmoid'或'relu'
    
    返回：
    A -- 当前层激活输出
    cache -- 包含线性缓存和激活缓存的元组
    """
    Z, linear_cache = linear_forward(A_prev, W, b)
    
    if activation == "sigmoid":
        A, activation_cache = sigmoid(Z)
    elif activation == "relu":
        A, activation_cache = relu(Z)
    
    cache = (linear_cache, activation_cache)
    return A, cache

def L_model_forward(X, parameters):
    """
    多层神经网络前向传播
    
    参数：
    X -- 输入数据，形状为 (input_size, m)
    parameters -- 包含各层参数的字典
    
    返回：
    AL -- 最后一层激活输出（预测值）
    caches -- 包含各层缓存的列表，用于反向传播
    """
    caches = []
    A = X
    L = len(parameters) // 2  # 参数数量除以2（每层有W和b）
    
    # 前L-1层使用ReLU激活
    for l in range(1, L):
        A_prev = A
        A, cache = linear_activation_forward(
            A_prev, 
            parameters[f'W{l}'], 
            parameters[f'b{l}'], 
            activation='relu'
        )
        caches.append(cache)
    
    # 最后一层使用Sigmoid激活（用于二分类）
    AL, cache = linear_activation_forward(
        A, 
        parameters[f'W{L}'], 
        parameters[f'b{L}'], 
        activation='sigmoid'
    )
    caches.append(cache)
    
    return AL, caches

# 测试前向传播
np.random.seed(42)
X = np.random.randn(3, 5)  # 3个特征，5个样本
parameters = initialize_parameters([3, 4, 2, 1])  # 3层网络：输入3，隐藏层4和2，输出1
AL, caches = L_model_forward(X, parameters)

print("输入X的形状:", X.shape)
print("输出AL的形状:", AL.shape)
print("预测值（前5个样本）:", AL[:, :5].flatten())
```

## 2. 反向传播的链式法则推导

### 2.1 链式法则基础

对于复合函数 $y = f(g(x))$，链式法则表示为：

$$dy/dx = dy/dg · dg/dx$$

在神经网络中，损失函数 $L$ 对参数 $θ$ 的梯度通过链式法则逐层计算：

$$∂ L/∂ θ[ˡ] = ∂ L/∂ a[L] · ∂ a[L]/∂ z[L] · ∂ z[L]/∂ a[L⁻¹] ·s ∂ a[ˡ]/∂ z[ˡ] · ∂ z[ˡ]/∂ θ[ˡ]$$

### 2.2 输出层的梯度计算

对于二分类问题，使用交叉熵损失函数：

$$L = -1/m Σ_ᵢ₌₁ᵐ ≤ft[ y⁽ⁱ⁾ (a[L]⁽ⁱ⁾) + (1-y⁽ⁱ⁾) (1-a[L]⁽ⁱ⁾) ]$$

其中 $a[L] = σ(z[L])$ 是Sigmoid激活函数的输出。

损失对 $z[L]$ 的梯度为：

$$∂ L/∂ z[L] = a[L] - y$$

**推导过程**：
1. Sigmoid函数的导数：$σ'(z) = σ(z)(1-σ(z))$
2. 交叉熵损失对 $a[L]$ 的导数：$∂ L/∂ a[L] = -y/a[L] + (1-y)/(1-a[L])$
3. 应用链式法则：
   $$∂ L/∂ z[L] = ∂ L/∂ a[L] · ∂ a[L]/∂ z[L] = ≤ft( -y/a[L] + (1-y)/(1-a[L]) ) · a[L](1-a[L]) = a[L] - y$$

### 2.3 隐藏层的梯度计算

对于第 $l$ 层（$l < L$），梯度计算需要从后向前传播：

1. 计算损失对 $z[ˡ]$ 的梯度：
   $$∂ L/∂ z[ˡ] = ∂ L/∂ a[ˡ] · σ'(z[ˡ])$$

2. 计算损失对 $W[ˡ]$ 和 $b[ˡ]$ 的梯度：
   $$∂ L/∂ W[ˡ] = ∂ L/∂ z[ˡ] · (a[ˡ⁻¹])ᵀ$$
   $$∂ L/∂ b[ˡ] = Σ_ᵢ₌₁ᵐ ∂ L/∂ z[ˡ]⁽ⁱ⁾$$

3. 计算损失对 $a[ˡ⁻¹]$ 的梯度（用于前一层）：
   $$∂ L/∂ a[ˡ⁻¹] = (W[ˡ])ᵀ · ∂ L/∂ z[ˡ]$$

### 2.4 Python实现：反向传播

```python
def sigmoid_backward(dA, activation_cache):
    """Sigmoid激活函数的反向传播"""
    Z = activation_cache
    s = 1 / (1 + np.exp(-Z))
    dZ = dA * s * (1 - s)
    return dZ

def relu_backward(dA, activation_cache):
    """ReLU激活函数的反向传播"""
    Z = activation_cache
    dZ = np.array(dA, copy=True)
    dZ[Z <= 0] = 0
    return dZ

def linear_backward(dZ, cache):
    """
    单层线性反向传播
    
    参数：
    dZ -- 损失对当前层线性输出Z的梯度
    cache -- 来自前向传播的缓存，包含(A_prev, W, b)
    
    返回：
    dA_prev -- 损失对上一层激活输出的梯度
    dW -- 损失对当前层权重的梯度
    db -- 损失对当前层偏置的梯度
    """
    A_prev, W, b = cache
    m = A_prev.shape[1]
    
    dW = np.dot(dZ, A_prev.T) / m
    db = np.sum(dZ, axis=1, keepdims=True) / m
    dA_prev = np.dot(W.T, dZ)
    
    return dA_prev, dW, db

def linear_activation_backward(dA, cache, activation):
    """
    单层线性+激活反向传播
    
    参数：
    dA -- 损失对当前层激活输出的梯度
    cache -- 包含线性缓存和激活缓存的元组
    activation -- 激活函数名称
    
    返回：
    dA_prev -- 损失对上一层激活输出的梯度
    dW -- 损失对当前层权重的梯度
    db -- 损失对当前层偏置的梯度
    """
    linear_cache, activation_cache = cache
    
    if activation == "relu":
        dZ = relu_backward(dA, activation_cache)
    elif activation == "sigmoid":
        dZ = sigmoid_backward(dA, activation_cache)
    
    dA_prev, dW, db = linear_backward(dZ, linear_cache)
    
    return dA_prev, dW, db

def compute_cost(AL, Y):
    """
    计算交叉熵损失
    
    参数：
    AL -- 预测值，形状为 (1, m)
    Y -- 真实标签，形状为 (1, m)
    
    返回：
    cost -- 交叉熵损失
    """
    m = Y.shape[1]
    
    # 避免log(0)的情况
    AL = np.clip(AL, 1e-15, 1 - 1e-15)
    
    cost = -np.sum(Y * np.log(AL) + (1 - Y) * np.log(1 - AL)) / m
    cost = np.squeeze(cost)
    
    return cost

def L_model_backward(AL, Y, caches):
    """
    多层神经网络反向传播
    
    参数：
    AL -- 最后一层激活输出（预测值）
    Y -- 真实标签
    caches -- 来自前向传播的缓存列表
    
    返回：
    grads -- 包含各层梯度的字典
    """
    grads = {}
    L = len(caches)
    m = AL.shape[1]
    Y = Y.reshape(AL.shape)
    
    # 初始化反向传播
    dAL = - (np.divide(Y, AL) - np.divide(1 - Y, 1 - AL))
    
    # 输出层反向传播（Sigmoid激活）
    current_cache = caches[L-1]
    grads[f"dA{L-1}"], grads[f"dW{L}"], grads[f"db{L}"] = linear_activation_backward(
        dAL, current_cache, activation="sigmoid"
    )
    
    # 隐藏层反向传播（ReLU激活）
    for l in reversed(range(L-1)):
        current_cache = caches[l]
        dA_prev_temp, dW_temp, db_temp = linear_activation_backward(
            grads[f"dA{l+1}"], current_cache, activation="relu"
        )
        grads[f"dA{l}"] = dA_prev_temp
        grads[f"dW{l+1}"] = dW_temp
        grads[f"db{l+1}"] = db_temp
    
    return grads

# 测试反向传播
np.random.seed(42)
Y = np.random.randint(0, 2, (1, 5))  # 5个样本的标签

# 计算损失
cost = compute_cost(AL, Y)
print("交叉熵损失:", cost)

# 反向传播
grads = L_model_backward(AL, Y, caches)

# 检查梯度形状
print("\n梯度形状检查:")
for key in grads:
    print(f"{key}: {grads[key].shape}")
```

## 3. 常见激活函数导数计算

### 3.1 Sigmoid函数

**函数定义**：
$$σ(z) = 1/(1 + e⁻ᶻ)$$

**导数推导**：
$${aligned} σ'(z) &= d/dz ≤ft( 1/(1 + e⁻ᶻ) ) &= e⁻ᶻ/((1 + e⁻ᶻ)²) &= 1/(1 + e⁻ᶻ) · e⁻ᶻ/(1 + e⁻ᶻ) &= σ(z) · (1 - σ(z)) {aligned}$$

**特性分析**：
- 输出范围：(0, 1)
- 梯度范围：(0, 0.25]
- 存在梯度消失问题：当 $|z|$ 较大时，梯度接近0

### 3.2 ReLU函数

**函数定义**：
$$ReLU(z) = max(0, z)$$

**导数**：
$$ReLU'(z) = {cases} 1 & if z > 0 0 & if z ≤ 0 {cases}$$

**特性分析**：
- 计算简单，无指数运算
- 缓解梯度消失问题（正区间梯度为1）
- 存在神经元死亡问题：负区间梯度为0

### 3.3 Leaky ReLU函数

**函数定义**：
$$LeakyReLU(z) = {cases} z & if z > 0 α z & if z ≤ 0 {cases}$$
其中 $α$ 是一个小的正数（通常取0.01）

**导数**：
$$LeakyReLU'(z) = {cases} 1 & if z > 0 α & if z ≤ 0 {cases}$$

### 3.4 Tanh函数

**函数定义**：
$$(z) = (eᶻ - e⁻ᶻ)/(eᶻ + e⁻ᶻ)$$

**导数推导**：
$${aligned} '(z) &= d/dz ≤ft( (eᶻ - e⁻ᶻ)/(eᶻ + e⁻ᶻ) ) &= ((eᶻ + e⁻ᶻ)² - (eᶻ - e⁻ᶻ)²)/((eᶻ + e⁻ᶻ)²) &= 1 - ²(z) {aligned}$$

**特性分析**：
- 输出范围：(-1, 1)
- 梯度范围：(0, 1]
- 相比Sigmoid，梯度消失问题有所缓解

## 4. 损失函数梯度示例

### 4.1 均方误差损失（MSE）

**函数定义**：
$$L(y, {y}) = 1/2m Σ_ᵢ₌₁ᵐ (y⁽ⁱ⁾ - {y}⁽ⁱ⁾)²$$

其中 ${y}⁽ⁱ⁾ = a[L]⁽ⁱ⁾$ 是神经网络的输出。

**梯度推导**：
$${∂ L}{∂ {y}⁽ⁱ⁾} = -1/m (y⁽ⁱ⁾ - {y}⁽ⁱ⁾)$$

向量形式：
$${∂ L}{∂ {y}} = -1/m (y - {y})$$

### 4.2 交叉熵损失（二分类）

**函数定义**：
$$L(y, {y}) = -1/m Σ_ᵢ₌₁ᵐ ≤ft[ y⁽ⁱ⁾ ({y}⁽ⁱ⁾) + (1-y⁽ⁱ⁾) (1-{y}⁽ⁱ⁾) ]$$

**梯度推导**：
$${∂ L}{∂ {y}⁽ⁱ⁾} = -1/m ≤ft( {y⁽ⁱ⁾}{{y}⁽ⁱ⁾} - {1-y⁽ⁱ⁾}{1-{y}⁽ⁱ⁾} )$$

当 ${y}⁽ⁱ⁾ = σ(z⁽ⁱ⁾)$ 时：
$$∂ L/∂ z⁽ⁱ⁾ = {y}⁽ⁱ⁾ - y⁽ⁱ⁾$$

### 4.3 交叉熵损失（多分类）

**函数定义**（使用Softmax）：
$$L(y, {y}) = -1/m Σ_ᵢ₌₁ᵐ Σ_ₖ₌₁K yₖ⁽ⁱ⁾ ({y}ₖ⁽ⁱ⁾)$$

其中 ${y}ₖ⁽ⁱ⁾ = (eᶻₖ⁽ⁱ⁾)/(Σ_ⱼ₌₁K eᶻⱼ⁽ⁱ⁾)$ 是Softmax函数的输出。

**梯度推导**：
$$∂ L/∂ zₖ⁽ⁱ⁾ = {y}ₖ⁽ⁱ⁾ - yₖ⁽ⁱ⁾$$

### 4.4 Python实现：损失函数梯度验证

```python
import numpy as np

def mse_loss(y_true, y_pred):
    """均方误差损失"""
    m = y_true.shape[1]
    loss = np.sum((y_true - y_pred)**2) / (2 * m)
    return loss

def mse_gradient(y_true, y_pred):
    """MSE损失的梯度"""
    m = y_true.shape[1]
    return -(y_true - y_pred) / m

def binary_crossentropy_loss(y_true, y_pred):
    """二分类交叉熵损失"""
    m = y_true.shape[1]
    # 避免log(0)
    y_pred = np.clip(y_pred, 1e-15, 1 - 1e-15)
    loss = -np.sum(y_true * np.log(y_pred) + (1 - y_true) * np.log(1 - y_pred)) / m
    return loss

def binary_crossentropy_gradient(y_true, y_pred):
    """二分类交叉熵损失的梯度"""
    m = y_true.shape[1]
    y_pred = np.clip(y_pred, 1e-15, 1 - 1e-15)
    grad = -(y_true / y_pred - (1 - y_true) / (1 - y_pred)) / m
    return grad

def softmax(z):
    """Softmax函数"""
    # 数值稳定性：减去最大值
    exp_z = np.exp(z - np.max(z, axis=0, keepdims=True))
    return exp_z / np.sum(exp_z, axis=0, keepdims=True)

def categorical_crossentropy_loss(y_true, y_pred):
    """多分类交叉熵损失"""
    m = y_true.shape[1]
    y_pred = np.clip(y_pred, 1e-15, 1)
    loss = -np.sum(y_true * np.log(y_pred)) / m
    return loss

def softmax_crossentropy_gradient(y_true, y_pred):
    """Softmax+交叉熵损失的梯度（合并计算，提高数值稳定性）"""
    m = y_true.shape[1]
    return (y_pred - y_true) / m

n_features = 5

z = np.random.randn(3, m)  # 逻辑值

print("MSE梯度形状:", mse_grad.shape)
print("MSE梯度（前3个样本）:", mse_grad[:, :3])

bce_value = binary_crossentropy_loss(y_true_binary, y_pred_binary)
bce_grad = binary_crossentropy_gradient(y_true_binary, y_pred_binary)

print("\n二分类交叉熵损失:", bce_value)
print("二分类交叉熵梯度形状:", bce_grad.shape)

# 数值梯度验证
def numerical_gradient(f, x, h=1e-5):
    """计算函数f在x处的数值梯度"""
    grad = np.zeros_like(x)
    
    # 遍历每个元素
    it = np.nditer(x, flags=['multi_index'], op_flags=['readwrite'])
    while not it.finished:
        idx = it.multi_index
        original_value = x[idx]
        
        # f(x + h)
        x[idx] = original_value + h
        fxh1 = f()
        
        # f(x - h)
        x[idx] = original_value - h
        fxh2 = f()
        
        # 梯度
        grad[idx] = (fxh1 - fxh2) / (2 * h)
        
        # 恢复原值
        x[idx] = original_value
        it.iternext()
    
    return grad
```

```python
        parameters[f'b{l}'] -= learning_rate * grads[f'db{l}']
```
