**题目1.2.2**：实现一个高效的环形缓冲区，用于存储实时传感器数据。

```python
import numpy as np

class RingBuffer:
    """
    高效环形缓冲区实现
    
    功能：
    - 固定大小，新数据覆盖旧数据
    - O(1) 时间复杂度的插入和读取
    - 支持批量获取最近N个数据点
    """
    
    def __init__(self, size, dtype=np.float32):
        self.size = size
        self.dtype = dtype
        self.buffer = np.zeros(size, dtype=dtype)
        self.pointer = 0
        self.filled = False
    
    def append(self, value):
        """添加单个数据点"""
        self.buffer[self.pointer] = value
        self.pointer = (self.pointer + 1) % self.size
        if self.pointer == 0:
            self.filled = True
    
    def extend(self, values):
        """批量添加数据"""
        values = np.asarray(values, dtype=self.dtype)
        n = len(values)
        
        if n >= self.size:
            # 如果数据量超过缓冲区大小，只保留最后size个
            self.buffer[:] = values[-self.size:]
            self.pointer = 0
            self.filled = True
            return
        
        remaining = self.size - self.pointer
        if remaining >= n:
            # 可以直接放入
            self.buffer[self.pointer:self.pointer + n] = values
            self.pointer += n
            if self.pointer == self.size:
                self.pointer = 0
                self.filled = True
        else:
            # 需要分割存放
            self.buffer[self.pointer:] = values[:remaining]
            self.buffer[:n - remaining] = values[remaining:]
            self.pointer = n - remaining
            self.filled = True
    
    def get_all(self):
        """获取所有有效数据"""
        if not self.filled:
            return self.buffer[:self.pointer]
        else:
            return np.concatenate([
                self.buffer[self.pointer:],
                self.buffer[:self.pointer]
            ])
    
    def get_recent(self, n):
        """获取最近的n个数据点"""
        if n >= self.size or not self.filled:
            return self.get_all()
        
        start = (self.pointer - n) % self.size
        if start < self.pointer:
            return self.buffer[start:self.pointer]
        else:
            return np.concatenate([
                self.buffer[start:],
                self.buffer[:self.pointer]
            ])
    
    def get_statistics(self):
        """获取统计信息"""
        data = self.get_all()
        return {
            'mean': np.mean(data),
            'std': np.std(data),
            'min': np.min(data),
            'max': np.max(data),
            'count': len(data)
        }

# 使用示例
if __name__ == '__main__':
    buffer = RingBuffer(size=1000)
    
    # 添加数据
    for i in range(1500):
        buffer.append(i % 100)
    
    # 获取统计信息
    stats = buffer.get_statistics()
    print(f"统计数据: {stats}")
    
    # 获取最近100个点
    recent = buffer.get_recent(100)
    print(f"最近100个点的平均值: {np.mean(recent):.2f}")
```

```python
def control_servo_thread(servo_id, duration=5):
    """多线程版本"""

    """测试多线程"""
    servos = ['servo_1', 'servo_2', 'servo_3', 'servo_4']
    threads = []
    results = []
    
    start = time.time()
    
    for servo_id in servos:
        t = threading.Thread(
            target=lambda sid=servo_id: results.append(
                control_servo_thread(sid)
            )
        )
        threads.append(t)
        t.start()
    
    for t in threads:
        t.join()
```

**答案要点**：
- **GIL影响**：同一时刻只有一个线程执行Python字节码，多线程在CPU密集型任务中无法并行

- **CPU密集型**（如数据处理）：必须使用多进程或C扩展
- **硬件控制最佳实践**：
  - 数据采集：多线程 + 队列（生产者-消费者）
  - 数据处理：多进程 + 共享内存
  - 实时控制：Cython/C扩展重写关键路径
  - 极端实时：使用RTOS或实时Linux内核

**题目1.3.2**：实现一个生产者-消费者模型，用于多传感器数据采集与处理。

```python
import threading
import queue
import time
import random
from collections import defaultdict
import numpy as np

class SensorDataProducer:
    """传感器数据生产者"""
    
    def __init__(self, sensor_id, sample_rate, data_queue, stop_event):
        self.sensor_id = sensor_id
        self.sample_rate = sample_rate
        self.data_queue = data_queue
        self.stop_event = stop_event
        self.thread = None
    
    def start(self):
        """启动生产者线程"""
        self.thread = threading.Thread(target=self._run, daemon=True)
        self.thread.start()
        print(f"传感器 {self.sensor_id} 生产者已启动，采样率: {self.sample_rate} Hz")
    
    def _run(self):
        """采集循环"""
        interval = 1.0 / self.sample_rate
        sample_count = 0
        
        while not self.stop_event.is_set():
            start_time = time.perf_counter()
            
            # 模拟读取传感器
            value = self._read_sensor()
            
            # 放入队列（带时间戳）
            try:
                self.data_queue.put({
                    'sensor_id': self.sensor_id,
                    'value': value,
                    'timestamp': time.time(),
                    'sample_count': sample_count
                }, timeout=0.1)
                sample_count += 1
            except queue.Full:
                print(f"警告：传感器 {self.sensor_id} 队列已满，丢弃数据")
            
            # 补偿延迟，保持采样率
            elapsed = time.perf_counter() - start_time
            if elapsed < interval:
                time.sleep(interval - elapsed)
    
    def _read_sensor(self):
        """读取传感器数据（模拟）"""
        # 实际代码替换为：
        # - serial.Serial().readline()
        # - GPIO.input(pin)
        # - smbus.SMBus().read_byte_data()
        return random.gauss(50, 10)  # 模拟传感器噪声

class SensorDataConsumer:
    """传感器数据消费者"""
    
    def __init__(self, data_queue, stop_event, batch_size=100):
        self.data_queue = data_queue
        self.stop_event = stop_event
        self.batch_size = batch_size
        self.thread = None
        self.buffer = defaultdict(list)
        self.callback = None
    
    def set_callback(self, callback):
        """设置数据处理回调"""
        self.callback = callback
    
    def start(self):
        """启动消费者线程"""
        self.thread = threading.Thread(target=self._run, daemon=True)
        self.thread.start()
        print("消费者已启动")
    
    def _run(self):
        """消费循环"""
        while not self.stop_event.is_set():
            try:
                # 获取数据（超时1秒）
                data = self.data_queue.get(timeout=1.0)
                
                sensor_id = data['sensor_id']
                self.buffer[sensor_id].append(data)
                
                # 批量处理
                if len(self.buffer[sensor_id]) >= self.batch_size:
                    batch = self.buffer[sensor_id][:self.batch_size]
                    self.buffer[sensor_id] = self.buffer[sensor_id][self.batch_size:]
                    
                    # 调用回调处理
                    if self.callback:
                        self.callback(sensor_id, batch)
                
                self.data_queue.task_done()
            
            except queue.Empty:
                # 超时，检查是否应该停止
                continue
            except Exception as e:
                print(f"消费者错误：{e}")
    
    def flush(self):
        """刷新剩余数据"""
        for sensor_id, data_list in self.buffer.items():
            if data_list and self.callback:
                self.callback(sensor_id, data_list)
        self.buffer.clear()

class SensorAcquisitionSystem:
    """多传感器采集系统"""
    
    def __init__(self, sensors_config):
        """
        初始化采集系统
        
        Args:
            sensors_config: 传感器配置列表
                [{'id': 'sensor_1', 'rate': 100}, ...]
        """
        self.sensors_config = sensors_config
        self.data_queue = queue.Queue(maxsize=10000)
        self.stop_event = threading.Event()
        self.producers = []
        self.consumer = None
        self.statistics = defaultdict(dict)
    
    def start(self):
        """启动整个系统"""
        print("启动多传感器采集系统...")
        
        # 创建并启动生产者
        for config in self.sensors_config:
            producer = SensorDataProducer(
                sensor_id=config['id'],
                sample_rate=config['rate'],
                data_queue=self.data_queue,
                stop_event=self.stop_event
            )
            producer.start()
            self.producers.append(producer)
        
        # 创建并启动消费者
        self.consumer = SensorDataConsumer(
            data_queue=self.data_queue,
            stop_event=self.stop_event,
            batch_size=100
        )
        self.consumer.set_callback(self._process_batch)
        self.consumer.start()
    
    def stop(self):
        """停止整个系统"""
        print("停止采集系统...")
        
        self.stop_event.set()
        
        # 等待生产者停止
        for producer in self.producers:
            if producer.thread:
                producer.thread.join(timeout=2)
        
        # 刷新消费者
        if self.consumer:
            self.consumer.flush()
        
        print("采集系统已停止")
    
    def _process_batch(self, sensor_id, batch):
        """处理一批数据"""
        values = [item['value'] for item in batch]
        timestamps = [item['timestamp'] for item in batch]
        
        # 计算统计信息
        stats = {
            'mean': np.mean(values),
            'std': np.std(values),
            'min': np.min(values),
            'max': np.max(values),
            'count': len(values),
            'first_timestamp': timestamps[0],
            'last_timestamp': timestamps[-1]
        }
        
        self.statistics[sensor_id] = stats
        
        # 打印（实际可存入数据库或发送到云平台）
        print(f"传感器 {sensor_id}: 均值={stats['mean']:.2f}, "
              f"标准差={stats['std']:.2f}, 样本数={stats['count']}")
    
    def get_statistics(self):
        """获取所有传感器的统计信息"""
        return dict(self.statistics)

# 使用示例
if __name__ == '__main__':
    # 传感器配置
    sensors = [
        {'id': 'accelerometer_x', 'rate': 200},  # 200 Hz
        {'id': 'accelerometer_y', 'rate': 200},
        {'id': 'accelerometer_z', 'rate': 200},
        {'id': 'gyroscope_x', 'rate': 200},
        {'id': 'temperature', 'rate': 10},      # 10 Hz
        {'id': 'pressure', 'rate': 50}         # 50 Hz
    ]
    
    # 创建并启动系统
    system = SensorAcquisitionSystem(sensors)
    system.start()
    
    try:
        # 运行10秒
        time.sleep(10)
        
        # 获取统计信息
        stats = system.get_statistics()
        print("\n最终统计信息：")
        for sensor_id, stat in stats.items():
            print(f"  {sensor_id}: {stat}")
    
    finally:
        system.stop()
```

```python
        self.interface_type = interface_type

        self.connected = False

    def connect(self):
        """建立连接"""
        pass
    
    @abstractmethod
    def disconnect(self):
        """断开连接"""
        pass

    def read(self, size=1):
        """读取数据"""
        pass
    
    @abstractmethod
    def write(self, data):
        """写入数据"""
        pass
    
    def __enter__(self):
        self.connect()
        return self
```

```python
# gpio_interrupts.py - GPIO中断处理

import RPi.GPIO as GPIO
import time
from collections import deque
import threading

class GPIOInterruptHandler:
    """GPIO中断处理器"""
    
    def __init__(self, debounce_ms=200, max_history=1000):
        """
        初始化中断处理器
        
        Args:
            debounce_ms: 防抖时间（毫秒）
            max_history: 最大历史记录数
        """
        self.debounce_ms = debounce_ms
        self.max_history = max_history
        self.callbacks = {}
        self.history = deque(maxlen=max_history)
        self.lock = threading.Lock()
        GPIO.setmode(GPIO.BCM)
    
    def register_interrupt(self, pin, edge='rising', callback=None, pull='up'):
        """
        注册中断
        
        Args:
            pin: GPIO引脚号
            edge: 触发边沿 ('rising', 'falling', 'both')
            callback: 回调函数，签名: callback(pin, timestamp)
            pull: 上拉/下拉 ('up', 'down', None)
        """
        # 设置引脚方向
        if pull == 'up':
            GPIO.setup(pin, GPIO.IN, pull_up_down=GPIO.PUD_UP)
        elif pull == 'down':
            GPIO.setup(pin, GPIO.IN, pull_up_down=GPIO.PUD_DOWN)
        else:
            GPIO.setup(pin, GPIO.IN)
        
        # 设置边沿检测
        if edge == 'rising':
            gpio_edge = GPIO.RISING
        elif edge == 'falling':
            gpio_edge = GPIO.FALLING
        else:
            gpio_edge = GPIO.BOTH
        
        # 注册回调
        self.callbacks[pin] = callback
        
        # 添加事件检测
        GPIO.add_event_detect(
            pin,
            gpio_edge,
            callback=self._interrupt_handler,
            bouncetime=self.debounce_ms
        )
        
        print(f"已注册中断: GPIO{pin}, 边沿={edge}, 防抖={self.debounce_ms}ms")
    
    def _interrupt_handler(self, pin):
        """中断处理函数（在中断上下文中调用）"""
        timestamp = time.time()
        
        # 记录历史（使用锁保护）
        with self.lock:
            self.history.append({
                'pin': pin,
                'timestamp': timestamp,
                'state': GPIO.input(pin)
            })
        
        # 调用用户回调
        if pin in self.callbacks and self.callbacks[pin]:
            try:
                self.callbacks[pin](pin, timestamp)
            except Exception as e:
                print(f"回调执行错误: {e}")
    
    def get_history(self, pin=None, since=None):
        """
        获取历史记录
        
        Args:
            pin: 指定引脚（None表示所有）
            since: 起始时间戳
        
        Returns:
            历史记录列表
        """
        with self.lock:
            if pin is not None:
                result = [item for item in self.history if item['pin'] == pin]
            else:
                result = list(self.history)
            
            if since is not None:
                result = [item for item in result if item['timestamp'] >= since]
            
            return result
    
    def unregister_interrupt(self, pin):
        """取消中断注册"""
        if pin in self.callbacks:
            GPIO.remove_event_detect(pin)
            del self.callbacks[pin]
            print(f"已取消中断: GPIO{pin}")
    
    def cleanup(self):
        """清理资源"""
        GPIO.cleanup()
        print("GPIO已清理")

# 实际应用：编码器读取
class RotaryEncoder:
    """旋转编码器"""
    
    def __init__(self, clk_pin, dt_pin, callback=None):
        """
        初始化编码器
        
        Args:
            clk_pin: 时钟引脚
            dt_pin: 数据引脚
            callback: 回调函数，签名: callback(direction, count)
        """
        self.clk_pin = clk_pin
        self.dt_pin = dt_pin
        self.callback = callback
        self.counter = 0
        self.last_state = 0
        
        # 创建中断处理器
        self.interrupt_handler = GPIOInterruptHandler()
        
        # 注册中断
        self.interrupt_handler.register_interrupt(
            clk_pin,
            edge='both',
            callback=self._clk_interrupt,
            pull='up'
        )
    
    def _clk_interrupt(self, pin, timestamp):
        """时钟中断处理"""
        clk_state = GPIO.input(self.clk_pin)
        dt_state = GPIO.input(self.dt_pin)
        
        if clk_state != self.last_state:
            self.last_state = clk_state
            
            if clk_state == 1:
                if dt_state == 0:
                    # 顺时针
                    self.counter += 1
                    direction = 1
                else:
                    # 逆时针
                    self.counter -= 1
                    direction = -1
                
                if self.callback:
                    self.callback(direction, self.counter)
    
    def get_count(self):
        """获取计数值"""
        return self.counter
    
    def reset(self):
        """重置计数"""
        self.counter = 0
    
    def cleanup(self):
        """清理资源"""
        self.interrupt_handler.unregister_interrupt(self.clk_pin)
        self.interrupt_handler.cleanup()
```

```python
import time
import ctypes

# ============== 方案1：CPU亲和性绑定 ==============
def bind_thread_to_core(thread_id, core_id):
    """
    将线程绑定到指定CPU核心
    
    Args:
        thread_id: 线程ID
        core_id: CPU核心ID
    """
    try:
        libc = ctypes.CDLL('libc.so.6')
        
        # 获取CPU集
        cpu_set_t = ctypes.c_ulonglong(1 << core_id)
        
        # 设置亲和性
        result = libc.sched_setaffinity(thread_id, 8, ctypes.pointer(ctypes.byref(cpu_set_t)))
        
        if result == 0:
            print(f"线程 {thread_id} 已绑定到核心 {core_id}")
        else:
            print(f"绑定失败: {result}")

def set_realtime_priority(priority=10):
    """
    设置实时优先级
    
    Args:
        priority: 优先级 (1-99, 数值越大优先级越高)
    """
    try:
        libc = ctypes.CDLL('libc.so.6')
        
        # 定义调度参数

class HighPrecisionTimer:

    def __init__(self):
        self.libc = ctypes.CDLL('libc.so.6')
        self.ts = ctypes.create_string_buffer(16)  # timespec结构体

        """获取单调时间"""
        self.libc.clock_gettime(1, self.ts)  # CLOCK_MONOTONIC = 1
        sec = ctypes.c_long.from_buffer(self.ts, 0).value
        nsec = ctypes.c_long.from_buffer(self.ts, 8).value
        return sec + nsec / 1e9

        sec = int(seconds)

        self.libc.clock_nanosleep(0, 0, self.ts, None)

class PIDController:
    """PID控制器"""
    
    def __init__(self, kp, ki, kd, setpoint=0):
        """
        初始化PID控制器
        
        Args:
            kp: 比例系数
            ki: 积分系数
            kd: 微分系数
            setpoint: 设定值
        """
        self.kp = kp
        self.ki = ki
        self.kd = kd
        self.setpoint = setpoint
        self.integral = 0
        self.last_error = 0
    
    def update(self, measurement, dt):
        """
        计算控制输出
        
        Args:
            measurement: 当前测量值
            dt: 时间间隔（秒）
        
        Returns:
            控制输出
        """
        error = self.setpoint - measurement
        
        # 比例项
        proportional = self.kp * error
        
        # 积分项
        self.integral += error * dt
        integral = self.ki * self.integral
        
        # 微分项
        derivative = self.kd * (error - self.last_error) / dt
        self.last_error = error
        
        # 计算输出
        output = proportional + integral + derivative
        
        return output
    
    def reset(self):
        """重置控制器"""
        self.integral = 0
        self.last_error = 0
```

```python
def calculate_crc(data):
    """
    计算CRC16-Modbus校验码
    
    Args:
        data: 字节数据
    
    Returns:
        CRC校验码（2字节）
    """
    crc = 0xFFFF
    for byte in data:
        crc ^= byte
        for _ in range(8):
            if crc & 0x0001:
                crc = (crc >> 1) ^ 0xA001
            else:
                crc >>= 1
    return struct.pack('<H', crc)

def verify_crc(data):

    验证CRC校验码
    
    Args:
        data: 包含CRC的字节数据
    
    Returns:
        bool: 校验是否通过
    """
    if len(data) < 2:
        return False
    received_crc = data[-2:]
    calculated_crc = calculate_crc(data[:-2])
    return received_crc == calculated_crc

class ModbusRTUMaster:
    """Modbus RTU主站"""
    
    def __init__(self, port, baudrate=9600, timeout=1):
        """
        初始化主站
        
        Args:
            port: 串口号 (如'/dev/ttyUSB0'或'COM3')
            baudrate: 波特率
            timeout: 超时时间（秒）
        """
        self.port = port
        self.baudrate = baudrate
        self.timeout = timeout
        self.serial = None
        self.lock = threading.Lock()
    
    def connect(self):
        """连接串口"""
        try:
            self.serial = serial.Serial(
                port=self.port,
                baudrate=self.baudrate,
                bytesize=8,
                parity=serial.PARITY_NONE,
                stopbits=serial.STOPBITS_ONE,
                timeout=self.timeout
            )
            print(f"Modbus主站已连接: {self.port}")
        except serial.SerialException as e:
            raise ConnectionError(f"串口连接失败: {e}")
    
    def disconnect(self):
        """断开连接"""
        if self.serial and self.serial.is_open:
            self.serial.close()
            print("Modbus主站已断开")
    
    def execute(self, request):
        """
        执行Modbus请求
        
        Args:
            request: ModbusRTURequest对象
        
        Returns:
            ModbusRTUResponse对象
        """
        with self.lock:
            if not self.serial or not self.serial.is_open:
                raise IOError("串口未连接")
            
            # 发送请求
            request_bytes = request.to_bytes()
            self.serial.write(request_bytes)
            self.serial.flush()
            
            # 读取响应
            response_bytes = self._read_response()
            
            # 解析响应
            response = ModbusRTUResponse.from_bytes(response_bytes)
            
            # 检查异常
            if response.is_exception:
                exc_code = ModbusExceptionCode(response.exception_code)
                raise ModbusException(exc_code)
            
            return response
    
    def _read_response(self):
        """读取响应数据"""
        response = bytearray()
        
        # 读取从站地址和功能码
        header = self.serial.read(2)
        if len(header) < 2:
            raise TimeoutError("读取响应超时")
        response.extend(header)
        
        # 根据功能码确定剩余数据长度
        function_code = header[1]
        
        if function_code & 0x80:
            # 异常响应：还有1字节（异常码）
            exc_byte = self.serial.read(1)
            if len(exc_byte) < 1:
                raise TimeoutError("读取异常码超时")
            response.extend(exc_byte)
        else:
            # 正常响应
            if function_code in [0x01, 0x02]:
                # 读线圈/离散输入：1字节长度 + N字节数据
                length_byte = self.serial.read(1)
                if len(length_byte) < 1:
                    raise TimeoutError("读取长度超时")
                response.extend(length_byte)
                
                data_length = length_byte[0]
                data_bytes = self.serial.read(data_length)
                if len(data_bytes) < data_length:
                    raise TimeoutError("读取数据超时")
                response.extend(data_bytes)
            
            elif function_code in [0x03, 0x04]:
                # 读寄存器：1字节长度 + 2*N字节数据
                length_byte = self.serial.read(1)
                if len(length_byte) < 1:
                    raise TimeoutError("读取长度超时")
                response.extend(length_byte)
                
                data_length = length_byte[0]
                data_bytes = self.serial.read(data_length)
                if len(data_bytes) < data_length:
                    raise TimeoutError("读取数据超时")
                response.extend(data_bytes)
            
            elif function_code in [0x05, 0x06]:
                # 写单个线圈/寄存器：4字节数据
                data_bytes = self.serial.read(4)
                if len(data_bytes) < 4:
                    raise TimeoutError("读取回写数据超时")
                response.extend(data_bytes)
            
            elif function_code in [0x0F, 0x10]:
                # 写多个线圈/寄存器：4字节数据
                data_bytes = self.serial.read(4)
                if len(data_bytes) < 4:
                    raise TimeoutError("读取回写数据超时")
                response.extend(data_bytes)
        
        # 读取CRC（2字节）
        crc_bytes = self.serial.read(2)
        if len(crc_bytes) < 2:
            raise TimeoutError("读取CRC超时")
        response.extend(crc_bytes)
        
        return bytes(response)

        request = ModbusRTURequest.write_single_register(
            slave_id, address, value
        )
        return self.execute(request)
    
    def write_multiple_registers(self, slave_id, start_addr, values):
        """写多个寄存器"""
        request = ModbusRTURequest.write_multiple_registers(
            slave_id, start_addr, values

# ============== Modbus RTU从站 ==============
class ModbusRTUSlave:
    """Modbus RTU从站"""
    
    def __init__(self, slave_id, port, baudrate=9600):
        """
        初始化从站
        
        Args:
            slave_id: 从站地址 (1-247)
            port: 串口号
            baudrate: 波特率
        """
        self.slave_id = slave_id
        self.port = port
        self.baudrate = baudrate
        self.serial = None
        self.running = False
        self.thread = None
        
        # 寄存器存储
        self.coils = [False] * 65536
        self.discrete_inputs = [False] * 65536
        self.holding_registers = [0] * 65536
        self.input_registers = [0] * 65536
    
    def connect(self):
        """连接串口"""
        try:
            self.serial = serial.Serial(
                port=self.port,
                baudrate=self.baudrate,
                bytesize=8,
                parity=serial.PARITY_NONE,
                stopbits=serial.STOPBITS_ONE,
                timeout=0.1
            )
            print(f"Modbus从站已连接: {self.port}, 地址={self.slave_id}")
        except serial.SerialException as e:
            raise ConnectionError(f"串口连接失败: {e}")
    
    def start(self):
        """启动从站"""
        self.running = True
        self.thread = threading.Thread(target=self._run_loop, daemon=True)
        self.thread.start()
        print("Modbus从站已启动")
    
    def stop(self):
        """停止从站"""
        self.running = False
        if self.thread:
            self.thread.join(timeout=2)
        print("Modbus从站已停止")
    
    def disconnect(self):
        """断开连接"""
        self.stop()
        if self.serial and self.serial.is_open:
            self.serial.close()
            print("Modbus从站已断开")
    
    def _run_loop(self):
        """运行循环"""
        buffer = bytearray()
        
        while self.running:
            try:
                # 读取数据
                data = self.serial.read(1)
                if len(data) == 0:
                    continue
                
                buffer.extend(data)
                
                # 检查是否接收完整帧
                if len(buffer) >= 6:  # 最小帧长度
                    # 尝试解析
                    try:
                        slave_id = buffer[0]
                        function_code = buffer[1]
                        
                        # 检查是否为广播地址或本站地址
                        if slave_id != 0 and slave_id != self.slave_id:
                            buffer.clear()
                            continue
                        
                        # 计算预期帧长度
                        expected_length = self._calculate_frame_length(function_code, buffer)
                        
                        if len(buffer) >= expected_length:
                            # 验证CRC
                            if verify_crc(bytes(buffer[:expected_length])):
                                # 处理请求
                                request_data = bytes(buffer[:expected_length-2])
                                response = self._process_request(request_data)
                                
                                # 发送响应（非广播）
                                if slave_id != 0:
                                    self.serial.write(response)
                                
                                buffer = buffer[expected_length:]
                            else:
                                # CRC错误，丢弃
                                buffer.clear()
                    
                    except:
                        buffer.clear()
            
            except Exception as e:
                print(f"从站运行错误: {e}")
                buffer.clear()
                time.sleep(0.1)
    
    def _calculate_frame_length(self, function_code, buffer):
        """计算帧长度"""
        if function_code & 0x80:
            # 异常响应：固定4字节
            return 4
        
        elif function_code == 0x01 or function_code == 0x02:
            # 读线圈/离散输入：2字节头部 + 1字节长度 + N字节数据 + 2字节CRC
            if len(buffer) < 3:
                return len(buffer) + 1
            return 6 + buffer[2]
        
        elif function_code == 0x03 or function_code == 0x04:
            # 读寄存器：2字节头部 + 1字节长度 + 2*N字节数据 + 2字节CRC
            if len(buffer) < 3:
                return len(buffer) + 1
            return 6 + buffer[2]
        
        elif function_code in [0x05, 0x06]:
            # 写单个：固定8字节
            return 8
        
        elif function_code in [0x0F, 0x10]:
            # 写多个：如果已读取字节数
            if len(buffer) >= 6:
                return 9 + buffer[6]
            return len(buffer) + 1
        
        return len(buffer) + 2
    
    def _process_request(self, request_data):
        """处理请求"""
        slave_id = request_data[0]
        function_code = request_data[1]
        data = request_data[2:]
        
        try:
            if function_code == ModbusFunctionCode.READ_COILS.value:
                # 读线圈
                start_addr, count = struct.unpack('>HH', data[:4])
                response_data = self._read_coils(start_addr, count)
            
            elif function_code == ModbusFunctionCode.READ_DISCRETE_INPUTS.value:
                # 读离散输入
                start_addr, count = struct.unpack('>HH', data[:4])
                response_data = self._read_discrete_inputs(start_addr, count)
            
            elif function_code == ModbusFunctionCode.READ_HOLDING_REGISTERS.value:
                # 读保持寄存器
                start_addr, count = struct.unpack('>HH', data[:4])
                response_data = self._read_holding_registers(start_addr, count)
            
            elif function_code == ModbusFunctionCode.READ_INPUT_REGISTERS.value:
                # 读输入寄存器
                start_addr, count = struct.unpack('>HH', data[:4])
                response_data = self._read_input_registers(start_addr, count)
            
            elif function_code == ModbusFunctionCode.WRITE_SINGLE_COIL.value:
                # 写单个线圈
                address, value = struct.unpack('>HH', data)
                response_data = self._write_single_coil(address, value)
            
            elif function_code == ModbusFunctionCode.WRITE_SINGLE_REGISTER.value:
                # 写单个寄存器
                address, value = struct.unpack('>HH', data)
                response_data = self._write_single_register(address, value)
            
            elif function_code == ModbusFunctionCode.WRITE_MULTIPLE_COILS.value:
                # 写多个线圈
                start_addr, count = struct.unpack('>HH', data[:4])
                byte_count = data[4]
                values = data[5:5+byte_count]
                response_data = self._write_multiple_coils(start_addr, count, values)
            
            elif function_code == ModbusFunctionCode.WRITE_MULTIPLE_REGISTERS.value:
                # 写多个寄存器
                start_addr, count = struct.unpack('>HH', data[:4])
                byte_count = data[4]
                values = struct.unpack(f'>{count}H', data[5:5+byte_count])
                response_data = self._write_multiple_registers(start_addr, values)
            
            else:
                # 不支持的功能码
                raise ModbusExceptionCode.ILLEGAL_FUNCTION
            
            # 构造响应
            pdu = bytes([slave_id, function_code]) + response_data
            return pdu + calculate_crc(pdu)
        
        except ModbusExceptionCode as exc_code:
            # 异常响应
            pdu = bytes([slave_id, function_code | 0x80, exc_code.value])
            return pdu + calculate_crc(pdu)
    
    def _read_coils(self, start_addr, count):
        """读线圈"""
        byte_count = (count + 7) // 8
        coils_data = [0] * byte_count
        
        for i in range(count):
            addr = start_addr + i
            if addr < len(self.coils):
                if self.coils[addr]:
                    coils_data[i // 8] |= 1 << (i % 8)
        
        return bytes([byte_count]) + bytes(coils_data)
    
    def _read_discrete_inputs(self, start_addr, count):
        """读离散输入"""
        byte_count = (count + 7) // 8
        inputs_data = [0] * byte_count
        
        for i in range(count):
            addr = start_addr + i
            if addr < len(self.discrete_inputs):
                if self.discrete_inputs[addr]:
                    inputs_data[i // 8] |= 1 << (i % 8)
        
        return bytes([byte_count]) + bytes(inputs_data)
    
    def _read_holding_registers(self, start_addr, count):
        """读保持寄存器"""
        byte_count = count * 2
        registers_data = []
        
        for i in range(count):
            addr = start_addr + i
            if addr < len(self.holding_registers):
                registers_data.append(self.holding_registers[addr])
        
        return bytes([byte_count]) + struct.pack(f'>{count}H', *registers_data)
    
    def _read_input_registers(self, start_addr, count):
        """读输入寄存器"""
        byte_count = count * 2
        registers_data = []
        
        for i in range(count):
            addr = start_addr + i
            if addr < len(self.input_registers):
                registers_data.append(self.input_registers[addr])
        
        return bytes([byte_count]) + struct.pack(f'>{count}H', *registers_data)
    
    def _write_single_coil(self, address, value):
        """写单个线圈"""
        if address < len(self.coils):
            self.coils[address] = (value == 0xFF00)
        return struct.pack('>HH', address, value)
    
    def _write_single_register(self, address, value):
        """写单个寄存器"""
        if address < len(self.holding_registers):
            self.holding_registers[address] = value
        return struct.pack('>HH', address, value)
    
    def _write_multiple_coils(self, start_addr, count, values):
        """写多个线圈"""
        for i in range(count):
            addr = start_addr + i
            if addr < len(self.coils):
                byte_index = i // 8
                bit_index = i % 8
                self.coils[addr] = bool(values[byte_index] & (1 << bit_index))
        
        return struct.pack('>HH', start_addr, count)
    
    def _write_multiple_registers(self, start_addr, values):
        """写多个寄存器"""
        for i, value in enumerate(values):
            addr = start_addr + i
            if addr < len(self.holding_registers):
                self.holding_registers[addr] = value
        
        return struct.pack('>HH', start_addr, len(values))
```

```python
# robot_kinematics.py - 机械臂运动学

import numpy as np
from math import sin, cos, atan2, sqrt, pi

class DHParameter:
    """DH参数类"""
    
    def __init__(self, theta, d, a, alpha):
        """
        初始化DH参数
        
        Args:
            theta: 关节角度（弧度）
            d: 连杆偏置（米）
            a: 连杆长度（米）
            alpha: 连杆扭转角（弧度）
        """
        self.theta = theta
        self.d = d
        self.a = a
        self.alpha = alpha
    
    def transformation_matrix(self):
        """计算变换矩阵"""
        ct = cos(self.theta)
        st = sin(self.theta)
        ca = cos(self.alpha)
        sa = sin(self.alpha)
        
        return np.array([
            [ct, -st*ca, st*sa, self.a*ct],
            [st, ct*ca, -ct*sa, self.a*st],
            [0, sa, ca, self.d],
            [0, 0, 0, 1]
        ])

class RobotArm:
    """机械臂类"""
    
    def __init__(self):
        """初始化机械臂"""
        # 6自由度机械臂的DH参数
        self.dh_params = [
            DHParameter(theta=0, d=0.1519, a=0, alpha=pi/2),       # 关节1
            DHParameter(theta=0, d=0, a=-0.24365, alpha=0),          # 关节2
            DHParameter(theta=0, d=0, a=-0.21325, alpha=0),          # 关节3
            DHParameter(theta=0, d=0.13105, a=0, alpha=pi/2),        # 关节4
            DHParameter(theta=0, d=-0.08535, a=0, alpha=-pi/2),       # 关节5
            DHParameter(theta=0, d=-0.0921, a=0, alpha=0)            # 关节6
        ]
        
        # 关节限制（弧度）
        self.joint_limits = [
            (-pi, pi),          # 关节1
            (-pi/2, pi/2),      # 关节2
            (-pi, pi),          # 关节3
            (-pi, pi),          # 关节4
            (-pi, pi),          # 关节5
            (-pi, pi)           # 关节6
        ]
    
    def forward_kinematics(self, joint_angles):
        """
        正运动学求解
        
        Args:
            joint_angles: 6个关节角度（弧度）
        
        Returns:
            末端执行器的位姿（4x4齐次变换矩阵）
        """
        if len(joint_angles) != 6:
            raise ValueError("需要6个关节角度")
        
        # 检查关节限制
        for i, (angle, limit) in enumerate(zip(joint_angles, self.joint_limits)):
            if not (limit[0] <= angle <= limit[1]):
                logger.warning(f"关节{i+1}角度{angle}超出限制{limit}")
        
        # 更新DH参数
        for i in range(6):
            self.dh_params[i].theta = joint_angles[i]
        
        # 计算末端执行器位姿
        T = np.eye(4)
        for dh in self.dh_params:
            T = T @ dh.transformation_matrix()
        
        return T
    
    def get_end_effector_position(self, joint_angles):
        """
        获取末端执行器位置
        
        Args:
            joint_angles: 6个关节角度（弧度）
        
        Returns:
            位置向量[x, y, z]
        """
        T = self.forward_kinematics(joint_angles)
        return T[:3, 3]
    
    def get_end_effector_orientation(self, joint_angles):
        """
        获取末端执行器姿态（旋转矩阵）
        
        Args:
            joint_angles: 6个关节角度（弧度）
        
        Returns:
            3x3旋转矩阵
        """
        T = self.forward_kinematics(joint_angles)
        return T[:3, :3]
    
    def inverse_kinematics(self, target_pose, initial_guess=None):
        """
        逆运动学求解（数值解法）
        
        Args:
            target_pose: 目标位姿（4x4齐次变换矩阵）
            initial_guess: 初始关节猜测值
        
        Returns:
            关节角度数组（弧度）
        """
        if initial_guess is None:
            initial_guess = np.zeros(6)
        
        # 使用雅可比伪逆法
        from scipy.optimize import minimize
        
        def cost_function(joint_angles):
            """代价函数"""
            current_pose = self.forward_kinematics(joint_angles)
            
            # 位置误差
            position_error = np.linalg.norm(
                current_pose[:3, 3] - target_pose[:3, 3]
            )
            
            # 姿态误差（旋转矩阵差异）
            orientation_error = np.linalg.norm(
                current_pose[:3, :3] - target_pose[:3, :3]
            )
            
            return position_error + orientation_error
        
        # 约束条件
        bounds = self.joint_limits
        
        # 优化
        result = minimize(
            cost_function,
            initial_guess,
            bounds=bounds,
            method='SLSQP'
        )
        
        if result.success:
            return result.x
        else:
            raise RuntimeError(f"逆运动学求解失败: {result.message}")
    
    def jacobian(self, joint_angles):
        """
        计算雅可比矩阵
        
        Args:
            joint_angles: 6个关节角度（弧度）
        
        Returns:
            6x6雅可比矩阵
        """
        epsilon = 1e-6
        jacobian = np.zeros((6, 6))
        
        # 当前末端位姿
        T_current = self.forward_kinematics(joint_angles)
        
        # 数值微分计算雅可比矩阵
        for i in range(6):
            # 扰动
            joint_perturbed = joint_angles.copy()
            joint_perturbed[i] += epsilon
            
            # 扰动后的位姿
            T_perturbed = self.forward_kinematics(joint_perturbed)
            
            # 计算变化率
            delta_pose = T_perturbed - T_current
            
            # 分离位置和姿态
            jacobian[:3, i] = delta_pose[:3, 3] / epsilon
            jacobian[3:, i] = (delta_pose[:3, :3] @ T_current[:3, :3].T).diagonal() / epsilon
        
        return jacobian
    
    def check_collision(self, joint_angles):
        """
        碰撞检测（简化版）
        
        Args:
            joint_angles: 关节角度
        
        Returns:
            bool: 是否发生碰撞
        """
        # 计算每个关节的位置
        T = np.eye(4)
        joint_positions = []
        
        for dh in self.dh_params:
            T = T @ dh.transformation_matrix()
            joint_positions.append(T[:3, 3])
        
        # 简化碰撞检测：检查关节间距离
        for i in range(len(joint_positions) - 1):
            for j in range(i + 1, len(joint_positions)):
                distance = np.linalg.norm(
                    joint_positions[i] - joint_positions[j]
                )
                
                # 假设最小安全距离为0.05米
                if distance < 0.05:
                    logger.warning(f"关节{i+1}和关节{j+1}距离太近")
                    return True
        
        # 检查地面碰撞
        for pos in joint_positions:
            if pos[2] < 0:
                logger.warning(f"关节位置低于地面: {pos}")
                return True
        
        return False

class TrajectoryPlanner:
    """轨迹规划器"""
    
    def __init__(self):
        """初始化轨迹规划器"""
        pass
    
    @staticmethod
    def linear_interpolation(start, end, num_points):
        """
        线性插值
        
        Args:
            start: 起始点
            end: 终点
            num_points: 插值点数
        
        Returns:
            插值点列表
        """
        return np.linspace(start, end, num_points)
    
    @staticmethod
    def cubic_spline_interpolation(waypoints, num_points_per_segment):
        """
        三次样条插值
        
        Args:
            waypoints: 路径点列表
            num_points_per_segment: 每段的插值点数
        
        Returns:
            插值点列表
        """
        from scipy.interpolate import CubicSpline
        
        # 提取x, y, z坐标
        points = np.array(waypoints)
        x = points[:, 0]
        y = points[:, 1]
        z = points[:, 2]
        
        # 参数化
        t = np.linspace(0, len(waypoints) - 1, len(waypoints))
        
        # 三次样条插值
        cs_x = CubicSpline(t, x)
        cs_y = CubicSpline(t, y)
        cs_z = CubicSpline(t, z)
        
        # 生成轨迹点
        t_new = np.linspace(0, len(waypoints) - 1, 
                           (len(waypoints) - 1) * num_points_per_segment + 1)
        
        trajectory = np.column_stack([
            cs_x(t_new),
            cs_y(t_new),
            cs_z(t_new)
        ])
        
        return trajectory
    
    @staticmethod
    def velocity_profile(distance, max_velocity, acceleration):
        """
        梯形速度曲线
        
        Args:
            distance: 总距离
            max_velocity: 最大速度
            acceleration: 加速度
        
        Returns:
            时间-位置曲线
        """
        # 计算加速时间
        t_acc = max_velocity / acceleration
        
        # 计算加速距离
        d_acc = 0.5 * acceleration * t_acc ** 2
        
        # 检查是否能达到最大速度
        if 2 * d_acc > distance:
            # 不能达到最大速度，三角形速度曲线
            t_total = 2 * sqrt(distance / acceleration)
            
            # 生成时间点
            t = np.linspace(0, t_total, 1000)
            
            # 计算位置
            position = np.zeros_like(t)
            for i, ti in enumerate(t):
                if ti < t_total / 2:
                    position[i] = 0.5 * acceleration * ti ** 2
                else:
                    t_decel = ti - t_total / 2
                    d_max = 0.5 * acceleration * (t_total / 2) ** 2
                    position[i] = d_max + max_velocity * t_decel - 0.5 * acceleration * t_decel ** 2
        else:
            # 梯形速度曲线
            d_const = distance - 2 * d_acc
            t_const = d_const / max_velocity
            t_total = 2 * t_acc + t_const
            
            # 生成时间点
            t = np.linspace(0, t_total, 1000)
            
            # 计算位置
            position = np.zeros_like(t)
            for i, ti in enumerate(t):
                if ti < t_acc:
                    # 加速阶段
                    position[i] = 0.5 * acceleration * ti ** 2
                elif ti < t_acc + t_const:
                    # 匀速阶段
                    d_acc_phase = 0.5 * acceleration * t_acc ** 2
                    t_const_phase = ti - t_acc
                    position[i] = d_acc_phase + max_velocity * t_const_phase
                else:
                    # 减速阶段
                    d_acc_phase = 0.5 * acceleration * t_acc ** 2
                    d_const_phase = max_velocity * t_const
                    t_decel = ti - t_acc - t_const
                    position[i] = d_acc_phase + d_const_phase + max_velocity * t_decel - 0.5 * acceleration * t_decel ** 2
        
        return position
```

```python
import numpy as np
import matplotlib.pyplot as plt

import random

# ============== 粒子滤波器 ==============
class ParticleFilter:
    """粒子滤波器"""
    
    def __init__(self, num_particles=100):
        """
        初始化粒子滤波器
        
        Args:
            num_particles: 粒子数量
        """
        self.num_particles = num_particles
        self.particles = np.zeros((num_particles, 3))  # [x, y, theta]
        self.weights = np.ones(num_particles) / num_particles
    
    def init_particles(self, x_range, y_range, theta_range):
        """
        初始化粒子分布
        
        Args:
            x_range: x范围 [min, max]
            y_range: y范围 [min, max]
            theta_range: 角度范围 [min, max]
        """
        self.particles[:, 0] = np.random.uniform(x_range[0], x_range[1], self.num_particles)
        self.particles[:, 1] = np.random.uniform(y_range[0], y_range[1], self.num_particles)
        self.particles[:, 2] = np.random.uniform(theta_range[0], theta_range[1], self.num_particles)
    
    def predict(self, motion, motion_noise):
        """
        预测步骤（运动更新）
        
        Args:
            motion: 运动指令 [v, w]（速度，角速度）
            motion_noise: 运动噪声标准差 [v_noise, w_noise]
        """
        v, w = motion
        v_noise, w_noise = motion_noise
        
        dt = 0.1  # 时间步长
        
        # 添加噪声
        v_noisy = v + np.random.normal(0, v_noise, self.num_particles)
        w_noisy = w + np.random.normal(0, w_noise, self.num_particles)
        
        # 更新粒子
        self.particles[:, 0] += v_noisy * dt * np.cos(self.particles[:, 2])
        self.particles[:, 1] += v_noisy * dt * np.sin(self.particles[:, 2])
        self.particles[:, 2] += w_noisy * dt
        
        # 角度归一化到[-pi, pi]
        self.particles[:, 2] = np.arctan2(np.sin(self.particles[:, 2]), np.cos(self.particles[:, 2]))
    
    def update(self, measurements, landmarks, measurement_noise):
        """
        更新步骤（观测更新）
        
        Args:
            measurements: 测量值 [range, bearing]列表
            landmarks: 地标位置 [[x1, y1], [x2, y2], ...]
            measurement_noise: 测量噪声标准差 [range_noise, bearing_noise]
        """
        range_noise, bearing_noise = measurement_noise
        
        # 计算每个粒子的似然
        for i in range(self.num_particles):
            particle = self.particles[i]
            likelihood = 1.0
            
            # 计算每个地标的测量似然
            for j, measurement in enumerate(measurements):
                landmark = landmarks[j]
                
                # 预测测量值
                dx = landmark[0] - particle[0]
                dy = landmark[1] - particle[1]
                pred_range = np.sqrt(dx**2 + dy**2)
                pred_bearing = np.arctan2(dy, dx) - particle[2]
                
                # 归一化角度
                pred_bearing = np.arctan2(np.sin(pred_bearing), np.cos(pred_bearing))
                
                # 计算似然（高斯分布）
                meas_range, meas_bearing = measurement
                
                # 距离似然
                range_diff = meas_range - pred_range
                likelihood_range = np.exp(-(range_diff**2) / (2 * range_noise**2))
                
                # 角度似然
                bearing_diff = meas_bearing - pred_bearing
                bearing_diff = np.arctan2(np.sin(bearing_diff), np.cos(bearing_diff))
                likelihood_bearing = np.exp(-(bearing_diff**2) / (2 * bearing_noise**2))
                
                likelihood *= likelihood_range * likelihood_bearing
            
            self.weights[i] = likelihood
        
        # 归一化权重
        self.weights /= np.sum(self.weights)
    
    def resample(self):
        """重采样"""
        # 计算累积权重
        cumsum = np.cumsum(self.weights)
        
        # 生成均匀随机数
        random_numbers = np.sort(np.random.uniform(0, 1, self.num_particles))
        
        # 重采样
        new_particles = []
        idx = 0
        
        for rand in random_numbers:
            while cumsum[idx] < rand:
                idx += 1
            new_particles.append(self.particles[idx])
        
        self.particles = np.array(new_particles)
        
        # 重置权重
        self.weights = np.ones(self.num_particles) / self.num_particles
    
    def estimate_pose(self):
        """估计机器人位姿（加权平均）"""
        weighted_x = np.sum(self.particles[:, 0] * self.weights)
        weighted_y = np.sum(self.particles[:, 1] * self.weights)
        
        # 角度使用圆周平均
        sin_theta = np.sum(np.sin(self.particles[:, 2]) * self.weights)
        cos_theta = np.sum(np.cos(self.particles[:, 2]) * self.weights)
        weighted_theta = np.arctan2(sin_theta, cos_theta)
        
        return np.array([weighted_x, weighted_y, weighted_theta])
    
    def get_particles(self):
        """获取所有粒子"""
        return self.particles.copy()
```

```python
        self.kinematics = robot_kinematics
        self.current_state = None
        self.target_command = None
        self.trajectory = []
        self.trajectory_index = 0
        self.running = False
    
    async def execute_command(self, command: RobotCommand):
        """
        执行命令
        
        Args:
            command: 机器人命令
        """
        self.target_command = command
        
        # 计算轨迹
        self.trajectory = self._plan_trajectory(command)
        self.trajectory_index = 0
        self.running = True
        
        # 执行轨迹
        while self.running and self.trajectory_index < len(self.trajectory):
            waypoint = self.trajectory[self.trajectory_index]
            
            # 检查碰撞
            if self.kinematics.check_collision(waypoint['joints']):
                raise CollisionException("检测到碰撞")
            
            # 检查关节限制
            for i, joint_angle in enumerate(waypoint['joints']):
                limits = self.kinematics.joint_limits[i]
                if not (limits[0] <= joint_angle <= limits[1]):
                    raise JointLimitException(
                        f"关节{i+1}角度{joint_angle}超出限制"
                    )
            
            # 更新状态
            self.current_state = RobotState(
                position=waypoint['position'],
                orientation=waypoint['orientation'],
                joints=waypoint['joints'],
                velocity=waypoint['velocity'],
                timestamp=datetime.now().timestamp()
            )
            
            self.trajectory_index += 1
            await asyncio.sleep(0.01)  # 100Hz控制周期
        
        self.running = False
    
    def _plan_trajectory(self, command: RobotCommand):
        """
        规划轨迹
        
        Args:
            command: 机器人命令
        
        Returns:
            轨迹点列表
        """
        # 逆运动学求解
        start_joints = np.zeros(6) if self.current_state is None else self.current_state.joints
        
        try:
            target_joints = self.kinematics.inverse_kinematics(
                command.target_pose,
                initial_guess=start_joints
            )
        except RuntimeError as e:
            raise RobotException(f"逆运动学求解失败: {e}")
        
        # 关节空间插值
        num_points = int(100)  # 100个轨迹点
        joint_trajectory = np.linspace(start_joints, target_joints, num_points)
        
        # 生成完整轨迹
        trajectory = []
        for i, joints in enumerate(joint_trajectory):
            # 正运动学
            T = self.kinematics.forward_kinematics(joints)
            
            # 计算速度（简单差分）
            if i > 0:
                velocity = (joints - joint_trajectory[i-1]) / 0.01
            else:
                velocity = np.zeros(6)
            
            trajectory.append({
                'joints': joints,
                'position': T[:3, 3],
                'orientation': T[:3, :3],
                'velocity': velocity
            })
        
        return trajectory

        """停止执行"""
        self.running = False
```

### 答案1.3.1：GIL影响分析

**GIL的作用**：
- 保证同一时刻只有一个线程执行Python字节码
- 防止多线程竞争导致的内存问题

**受影响的场景**：
- CPU密集型任务：多线程无法并行
- I/O密集型任务：影响较小（I/O时释放GIL）

**解决方案对比**：

| 方案 | 适用场景 | 优点 | 缺点 |
|------|---------|------|------|
| 多线程 | I/O密集型 | 简单，共享内存 | 受GIL限制 |
| 多进程 | CPU密集型 | 真正并行 | 通信开销大 |
| C扩展 | 关键路径 | 高性能 | 开发复杂 |

**最佳实践**：
